/* [ZwrocNumerTypuKoduEAN] */

CREATE FUNCTION [CDN].[ZwrocNumerTypuKoduEAN](@cEAN varchar(40))
RETURNS tinyint
AS
BEGIN
	declare @RetVal tinyint;
	if(LEN(@cEAN) = 4 AND PATINDEX('%[^0-9]%',@cEAN)=0)
		set @RetVal = 2
	else if (LEN(@cEAN) = 13)
		if (CDN.CyfraKontrolnaEAN(@cEAN) = @cEAN)
			set @RetVal = 1
		else
			set @RetVal = 3
	else if (LEN(@cEAN)=8)
		if (CDN.CyfraKontrolnaEAN(@cEAN) = @cEAN)
			set @RetVal = 0
		else
			set @RetVal = 3
	else				
       set @RetVal = 3
       
	return @RetVal;      
END
