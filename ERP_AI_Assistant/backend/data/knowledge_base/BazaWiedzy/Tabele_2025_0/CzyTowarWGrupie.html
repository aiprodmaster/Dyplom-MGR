/* [CzyTowarWGrupie] */

CREATE Function [CDN].[CzyTowarWGrupie](@TwrNumer INT, @TwGNumer INT)
Returns bit
As
/***************************************************************/
/* Funkcja sprawdza czy towar o podanym gid numerze (@TwrNumer)*/ 
/* znajduje się w grupie o podanym gid numerze (@TwGNumer)	   */ 
/*															   */
/***************************************************************/
Begin

		Declare @Result bit
        Set @Result = 0

		IF EXISTS
		(
				select * from CDN.TwrLinki 
				where TwL_GIDTyp = 16 AND TwL_GIDNumer = @TwRNumer AND TwL_GRONumer = @TwGNumer
        )SELECT @Result = 1 ELSE SELECT @Result = 0
        
        Return @Result

End
