/* TraNag_UsunZD */

CREATE TRIGGER CDN.TraNag_UsunZD
  ON CDN.TraNag
  FOR DELETE
AS
SET NOCOUNT ON
  
  DELETE FROM cdn.PodElemDok 
  WHERE 
	EXISTS(SELECT 1 FROM deleted d
					left join cdn.TraNag trn on d.TrN_VatZDPeDNumer=trn.TrN_VatZDPeDNumer and d.TrN_VatZDPeDLp=trn.TrN_VatZDPeDLp and d.TrN_GIDNumer<>trn.trn_gidnumer and trn.TrN_VatZDPeDNumer<>0
			WHERE d.TrN_VatZDPeDNumer < 0 and d.TrN_VatZDPeDNumer=PED_PONNumer and trn.TRN_GIDNumer is null)


SET NOCOUNT OFF
