/* AtrybutyWartosci_CzasModyfikacji */

CREATE TRIGGER CDN.AtrybutyWartosci_CzasModyfikacji
  ON CDN.AtrybutyWartosci
  FOR INSERT,UPDATE
AS
SET NOCOUNT ON
    UPDATE CDN.AtrybutyKlasy SET AtK_CzasModyfikacji = DATEDIFF(s,convert(datetime,'1990-01-01',120),getdate()) 
    FROM inserted WHERE AtK_ID = inserted.AtW_AtkID
SET NOCOUNT OFF
