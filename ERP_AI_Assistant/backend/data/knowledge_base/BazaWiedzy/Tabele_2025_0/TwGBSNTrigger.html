/* TwrGrupy_BstHandler */

CREATE TRIGGER CDN.TwrGrupy_BstHandler
  ON CDN.TwrGrupy
  FOR DELETE
AS
SET NOCOUNT ON
	UPDATE CDN.BstNag SET BsN_TwgTyp = -16,BsN_TwgNumer = 0, BsN_TwGGroTyp = 0, BsN_TwGGroNumer = 0 FROM 
		deleted JOIN CDN.BstNag ON BsN_TwgTyp = deleted.TwG_GIDTyp AND BsN_TwgNumer = deleted.TwG_GIDNumer AND BsN_TwGGroTyp = deleted.TwG_GROTyp AND BsN_TwGGroNumer = deleted.TwG_GRONumer

SET NOCOUNT OFF
