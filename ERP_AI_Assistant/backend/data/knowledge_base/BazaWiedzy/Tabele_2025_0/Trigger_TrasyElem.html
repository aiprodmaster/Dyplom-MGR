/* [TrasyElem_DeletePrzenumeruj] */

CREATE TRIGGER [CDN].[TrasyElem_DeletePrzenumeruj]
   ON  [CDN].[TrasyElem]
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.TrasyElem SET TsE_Lp = E.TsE_Lp - 1
FROM CDN.TrasyElem E
INNER JOIN Deleted D ON E.TsE_TsNId = D.TsE_TsNId
WHERE D.TsE_Lp < E.TsE_Lp

SET NOCOUNT OFF
END
