/* ProdMaxKoniecPauzDlaDaty */

CREATE FUNCTION CDN.ProdMaxKoniecPauzDlaDaty (@PCZId INT,@TerminStart INT, @PPYId INT = 0)
RETURNS INT
AS
BEGIN
	DECLARE @MaxKoniec INT = 0
	SELECT @MaxKoniec = COALESCE((SELECT MIN(PPY_TerminStart) - 1
        FROM CDN.ProdOperacjePauzy
        WHERE PPY_PCZId =@PCZId AND PPY_Id <> @PPYId AND PPY_TerminStart > @TerminStart),0)
	RETURN @MaxKoniec
END
