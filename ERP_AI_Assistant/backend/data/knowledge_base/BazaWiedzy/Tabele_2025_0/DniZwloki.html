/* DniZwloki */

create function CDN.DniZwloki( @DataRoz int, @Termin int, @Today int, @Rozliczony int )
returns int
as
begin
  declare @retval int;
  if @Rozliczony = 0
    set @retval = @Today - @Termin;
  else
    set @retval = isnull(@DataRoz,0) - @Termin
  if @retval < 0
    set @retval = 0
  return @retval;
end
