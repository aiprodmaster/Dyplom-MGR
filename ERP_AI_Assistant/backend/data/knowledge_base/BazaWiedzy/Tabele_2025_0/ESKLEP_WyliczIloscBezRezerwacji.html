/* ESKLEP_WyliczIloscBezRezerwacji */

CREATE FUNCTION CDN.ESKLEP_WyliczIloscBezRezerwacji(@str varchar(255),@uwzglRez tinyint)
RETURNS 
DECIMAL(15,4)
AS
BEGIN
DECLARE @wartosc DECIMAL(15,4)
	set @wartosc = CAST(CDN.SubstringEx(@str,':',1) AS DECIMAL(15,4))-CASE WHEN @uwzglRez = 1 THEN CAST(CDN.SubstringEx(@str,':',3) AS DECIMAL(15,4)) ELSE 0 END
	IF @wartosc < 0  set @wartosc = 0
	return @wartosc
END
