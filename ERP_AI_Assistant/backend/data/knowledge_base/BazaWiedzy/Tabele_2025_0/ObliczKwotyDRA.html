/* [ObliczKwotyDRA] */

CREATE PROCEDURE [CDN].[ObliczKwotyDRA] 
	@KWOTA Varchar(20), 
	@WYNIK DECIMAL(15,2) OUTPUT
AS
BEGIN
SET NOCOUNT OFF
   DECLARE @l INT
   SET @l = LEN(@KWOTA)
   IF @l=0 OR @KWOTA is null
   BEGIN
      SET @WYNIK = CONVERT (DECIMAL(15,2), '0')
      RETURN
   END
   DECLARE @licznik VARCHAR(10)
   DECLARE @mianownik VARCHAR(2)
   DECLARE @skwota VARCHAR(20)
   SET @licznik = SUBSTRING (@KWOTA, 1, @l-2)
   SET @mianownik = SUBSTRING (@KWOTA, @l-1, 2)
   SET @skwota = @licznik+'.'+@mianownik
   SET @WYNIK = CONVERT (DECIMAL(15,2), @skwota)

END
