/* PRMRefundacjaRozliczona */

CREATE FUNCTION CDN.PRMRefundacjaRozliczona
(
	@lPrm			INTEGER
)
RETURNS DECIMAL(20,2)
AS 
BEGIN
	DECLARE @Ret DECIMAL(20,2)
	SELECT @Ret = SUM(TrN_NettoR+TrN_NettoP) FROM CDN.TraNag (NOLOCK)
		JOIN CDN.DokZwiazane ON DOZ_ZrdTyp = 5104 AND DOZ_ZRDNumer = @lPrm AND DOZ_DOKTyp = TrN_GIDTyp AND DOZ_DOKNumer = TrN_GIDNumer
	RETURN @Ret
END
