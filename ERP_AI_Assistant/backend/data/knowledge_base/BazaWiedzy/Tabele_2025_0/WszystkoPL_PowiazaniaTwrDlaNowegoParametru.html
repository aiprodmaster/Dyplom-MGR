/* [WszystkoPl_PowiazaniaTwrDlaNowegoParametru] */

CREATE  PROCEDURE [CDN].[WszystkoPl_PowiazaniaTwrDlaNowegoParametru]
@WszystkoPLId  INT -- ID z wszystko.pl
AS
SET NOCOUNT ON

INSERT INTO cdn.KgtWartoscParametrDlaTowaru (KWPT_TwrNumer, KWPT_ParametrKategoriiId)
        SELECT TAP_ObiNumer, KPK_Id
        FROM cdn.KgtParametryKategorii
		JOIN cdn.twraplikacje on TAP_WszystkoPlKategoria = @WszystkoPLId
		LEFT JOIN cdn.KgtWartoscParametrDlaTowaru ON KPK_Id = KWPT_ParametrKategoriiId
        WHERE KPK_KategoriaId =  TAP_WszystkoPlKategoria AND KWPT_ParametrKategoriiId is null
