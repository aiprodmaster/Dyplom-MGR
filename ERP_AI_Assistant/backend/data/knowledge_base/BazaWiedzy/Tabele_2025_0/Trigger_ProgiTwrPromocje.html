/* [ProgiTwrPromocje_InsertHandlerTwG] */

CREATE TRIGGER [CDN].[ProgiTwrPromocje_InsertHandlerTwG]
ON [CDN].[ProgiTwrPromocje]
FOR INSERT
AS
BEGIN
	UPDATE PTP SET PTP_Wartosc = PTPG.PTP_Wartosc
	FROM CDN.ProgiTwrPromocje PTP
	INNER JOIN inserted on inserted.PTP_PRPId = PTP.PTP_PRPId and inserted.PTP_TPRId = PTP.PTP_TPRId
	INNER JOIN CDN.ProgiPromocje on PRP_ID = PTP.PTP_PRPId
	INNER JOIN CDN.TwrPromocje TPTwr on TPTwr.TPR_Id = PTP.PTP_TPRId
	INNER JOIN CDN.TwrPromocje TPTwg on TPTwg.TPR_TwrNumer = TPTwr.TPR_TwgNumer and TPTwg.TPR_TwrTyp = TPTwr.TPR_TwgTyp and TPTwg.TPR_PrmId = TPTwr.TPR_PrmId
	INNER JOIN CDN.ProgiTwrPromocje PTPG on PTPG.PTP_TPRId = TPTwg.TPR_Id	
END
