/* AiZapiszTerminPlatnosci */

Create Procedure CDN.AiZapiszTerminPlatnosci(@TrPTyp smallint, @TrPNumer int, @TrPLp smallint, @Opoznienie int)
		as
		begin
			update CDN.TraPlat set TrP_SpodziewanyTermin=TrP_Termin + @Opoznienie where TrP_GIDTyp=@TrPTyp and TrP_GIDNumer=@TrPNumer and TrP_GIDLp=@TrPLp;
		end
