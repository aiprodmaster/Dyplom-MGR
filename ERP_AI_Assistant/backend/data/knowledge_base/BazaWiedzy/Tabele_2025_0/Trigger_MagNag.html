/* [MagNag_LastMod_OFFL] */

CREATE TRIGGER [CDN].[MagNag_LastMod_OFFL]
ON [CDN].[MagNag]
FOR UPDATE
AS
BEGIN
  IF UPDATE(MaN_ZewnetrznySys) OR UPDATE(MaN_ZewnetrzneId)
  BEGIN
    UPDATE CDN.MagNag SET MaN_LastMod = datediff(s,convert(datetime,'1990-01-01',120),getdate())
    FROM CDN.MagNag T JOIN inserted I ON T.MaN_GIDNumer = I.MaN_GIDNumer
  END
END
