/* ECODUltimateCustomer */

CREATE FUNCTION CDN.ECODUltimateCustomer (@GIDTyp SMALLINT,@GIDNumer INT,@GIDLp SMALLINT)
RETURNS varchar(13)
AS
BEGIN
		DECLARE @UltimateCustomer VARCHAR(255)
		SELECT @UltimateCustomer = ISNULL(Knt_GLN,'')
		FROM CDN.AtrybutyKlasy 
		LEFT OUTER JOIN CDN.Atrybuty ON Atr_AtkID = Atk_ID
		LEFT OUTER JOIN CDN.KntKarty ON Atr_Wartosc = Knt_Akronim
		WHERE Atk_Nazwa = 'UltimateCustomer' AND Atr_ObiTyp = @GIDTyp AND Atr_ObiNumer = @GIDNumer AND Atr_ObiLp = @GIDLp
		
        RETURN @UltimateCustomer
END
