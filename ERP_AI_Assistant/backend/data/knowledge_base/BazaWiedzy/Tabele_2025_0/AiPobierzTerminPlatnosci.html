/* AiPobierzTerminPlatnosci */

Create Procedure CDN.AiPobierzTerminPlatnosci(@TrPTyp smallint, @TrPNumer int, @TrPLp smallint)
		as
		begin
			select TTV_Data, TTV_Termin from CDN.TraPlat inner join CDN.TrpTrnView on TTV_GIDNumer=TrP_GIDNumer and TTV_GIDTyp=TrP_GIDTyp and TTV_GIDLp=TrP_GIDLp 
            where TrP_GIDTyp = @TrPTyp and TrP_GIDNumer = @TrPNumer and TrP_GIDLp = @TrPLp;
		end
