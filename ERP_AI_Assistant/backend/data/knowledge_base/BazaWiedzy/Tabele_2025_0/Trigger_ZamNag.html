/* ZamNag_LastModTStamp */

CREATE TRIGGER CDN.ZamNag_LastModTStamp
ON CDN.ZamNag
FOR UPDATE
AS
SET NOCOUNT ON 
	IF Context_Info() = 0x15ff2808
	RETURN

	UPDATE CDN.ZamNag SET ZaN_LastMod=DateDiff(second,convert(datetime,'1990-01-01',120),getdate())
	FROM CDN.ZamNag Z JOIN inserted I ON Z.ZaN_GIDNumer=I.ZaN_GIDNumer
SET NOCOUNT OFF
