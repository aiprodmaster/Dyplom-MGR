/* StaKarty_Usuwanie */

CREATE TRIGGER CDN.StaKarty_Usuwanie
  ON CDN.StaKarty
  FOR DELETE
AS
SET NOCOUNT ON
    -- obsługa kaskadowego kasowania wierszy z tabeli CDN.AtrybutyUprawnienia
    DELETE CDN.AtrybutyUprawnienia
    FROM CDN.AtrybutyUprawnienia, deleted
    WHERE
          deleted.Sta_GIDNumer = -CDN.AtrybutyUprawnienia.AtU_OpeNumer
		AND (CDN.AtrybutyUprawnienia.AtU_OpeNumer<0)
SET NOCOUNT OFF
