/* [GratisyPromocje_DeletePrzenumeruj] */

CREATE TRIGGER [CDN].[GratisyPromocje_DeletePrzenumeruj]
   ON  [CDN].[GratisyPromocje]
   FOR DELETE
AS
BEGIN

UPDATE CDN.GratisyPromocje SET GPR_Pozycja = GP.GPR_Pozycja - 1
FROM CDN.GratisyPromocje GP
INNER JOIN Deleted D ON GP.GPR_PrmId = D.GPR_PrmId AND GP.GPR_PrPID = D.GPR_PrPID
WHERE D.GPR_Pozycja < GP.GPR_Pozycja

END
