/* [Rejestry_CzasModyfikacji] */

CREATE TRIGGER [CDN].[Rejestry_CzasModyfikacji]
ON [CDN].[Rejestry]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.kar_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.Rejestry N join inserted I on N.KAr_GIDTyp = I.KAr_GIDTyp and N.KAr_GIDNumer = I.kar_GIDNumer

	SET NOCOUNT OFF
END
