/* AktualizujIstotneOpcje */

CREATE PROCEDURE CDN.AktualizujIstotneOpcje @IDProcedury INT, @IDKontrolki INT, @IDOperatora INT, @sNazwaKontrolki VARCHAR(50), @sNazwaOpcji VARCHAR(50)
AS
	IF EXISTS(SELECT * FROM CDN.IstotneOpcje WHERE IOP_IDProcedury=@IDProcedury AND IOP_NazwaKontrolki=@sNazwaKontrolki AND IOP_IDOperatora = @IDOperatora AND IOP_NazwaOpcji = @sNazwaOpcji)
		UPDATE CDN.IstotneOpcje SET IOP_Licznik += 1 WHERE IOP_IDProcedury=@IDProcedury AND IOP_NazwaKontrolki=@sNazwaKontrolki AND IOP_IDOperatora = @IDOperatora AND IOP_NazwaOpcji = @sNazwaOpcji
	ELSE
		INSERT INTO CDN.IstotneOpcje (IOP_IDProcedury,IOP_NazwaKontrolki,IOP_NazwaOpcji,IOP_IDOperatora,IOP_Licznik) VALUES (@IDProcedury,@sNazwaKontrolki,@sNazwaOpcji,@IDOperatora,1)
