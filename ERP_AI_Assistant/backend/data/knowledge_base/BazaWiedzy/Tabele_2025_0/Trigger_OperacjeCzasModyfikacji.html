/* [Operacje_CzasModyfikacji] */

CREATE TRIGGER [CDN].[Operacje_CzasModyfikacji]
ON [CDN].[Operacje]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.kao_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.Operacje N join inserted I on N.KAO_GIDTyp = I.KAO_GIDTyp and N.KAO_GIDLp = I.kao_GIDLp

	SET NOCOUNT OFF
END
