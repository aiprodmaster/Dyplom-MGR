/* StanyTowaruWOddzialach */

CREATE PROCEDURE CDN.StanyTowaruWOddzialach (@Twr_GIDNumer INT)
AS
SELECT TwZ_MagNumer Id, MAX(MAG_Kod +' ('+MAG_Miasto + ', '+MAG_Ulica+', '+MAG_Adres+', '+MAG_Opis+')') Magazyn, SUM(TwZ_IlSpr) Ilosc 
FROM CDN.TwrZasoby JOIN CDN.Magazyny ON TwZ_MagNumer=MAG_GIDNumer
WHERE TwZ_TwrNumer=@Twr_GIDNumer
GROUP BY TwZ_MagNumer
