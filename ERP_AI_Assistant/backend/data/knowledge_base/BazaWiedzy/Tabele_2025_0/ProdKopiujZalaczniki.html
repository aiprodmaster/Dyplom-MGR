/* ProdKopiujZalaczniki */

CREATE PROCEDURE CDN.ProdKopiujZalaczniki @ZrdTyp INT, @ZrdNumer INT, @DocTyp INT, @DocNumer INT
as
begin
	INSERT INTO CDN.DaneObiekty (DAO_DABId, DAO_ObiTyp, DAO_ObiNumer, DAO_ObiLp, DAO_Domyslna, DAO_Blokada, DAO_PPPrawa, DAO_PKPrawa, DAO_eSklep, DAO_iMall, 
	DAO_MobSpr, DAO_Systemowa, DAO_Retail, DAO_Pozycja, DAO_ObiSubLp,DAO_BI,DAO_WMSZarzadzanie,DAO_WMSMagazynier) 
	SELECT DAO_DABId, @DocTyp, @DocNumer, DAO_ObiLp, DAO_Domyslna, DAO_Blokada,DAO_PPPrawa,DAO_PKPrawa,DAO_eSklep,DAO_iMall,DAO_MobSpr,DAO_Systemowa, 
	DAO_Retail,	DAO_Pozycja,DAO_ObiSubLp,DAO_BI,DAO_WMSZarzadzanie,DAO_WMSMagazynier 
	FROM CDN.DaneObiekty WHERE DAO_ObiTyp=@ZrdTyp and DAO_ObiNumer=@ZrdNumer		
end
