/* [WszystkoPL_AktualizujDabLink] */

CREATE  PROCEDURE [CDN].[WszystkoPL_AktualizujDabLink]
@ObiNumer  INT, --   	
@Link  VARCHAR(2048), --   
@DabID INT
AS
SET NOCOUNT ON

UPDATE cdn.DaneBinarne SET DAB_WszystkoPlLink = @Link from cdn.danebinarne
join cdn.DaneObiekty on dao_dabid = dab_id WHERE DAO_WszystkoPl = 1 AND DAB_Archiwalny = 0 AND dao_obityp = 16 AND Dab_Id = @DabID AND Dao_ObiNumer = @Obinumer

SET NOCOUNT OFF
