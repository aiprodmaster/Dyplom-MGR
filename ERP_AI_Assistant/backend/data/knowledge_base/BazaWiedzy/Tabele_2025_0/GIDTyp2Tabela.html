/* GIDTyp2Tabela */

create function CDN.GIDTyp2Tabela( @GIDTyp int)
returns int
as
begin
  declare @retval int;

  set @retval = (select case
    when @GIDTyp IN (0)
      then 0
    when @GIDTyp IN (7684)
      then 28 --preliminarz
    when @GIDTyp IN (1601,1089,1605,1093)
      then 2 --magnag
    when @GIDTyp IN (800)
      then 3  --raporty
    when @GIDTyp IN (784)
      then 4  --zapisy
    when @GIDTyp IN (2752)
      then 5 --ksonag
    when @GIDTyp IN (2048,2064,2080,2096,2112,2128,2144,2160,2176,2192,2224,2288,2304,6208)
      then 6 --srthistnag
    when @GIDTyp IN (432)
      then 7 --dziennik
    when @GIDTyp IN (433)
      then 8 --rozrachunki
    when @GIDTyp IN (2976,2977)
      then 9 --piknag
    when @GIDTyp IN (3376,3377,3378,3379,3386,3387)
      then 10 --sadnag
    when @GIDTyp IN (3344)
      then 11 --impnag
    when @GIDTyp IN (960,8832,8960,9344,9472,10880,11008,11392,11520,9473)
      then 12 --zamnag
    when @GIDTyp IN (4144)
      then 13 --memnag
    when @GIDTyp IN (4192,4208,4240)
      then 14 --zlcnag
    when @GIDTyp IN (2832,2833)
      then 15 --uponag
    when @GIDTyp IN (4288)
      then 16 --przeszacnag
    when @GIDTyp IN (4320)
      then 17 --dokdefinicje
    when @GIDTyp IN (4336)
      then 18 --zlecwewnag
    when @GIDTyp IN (7168)
      then 19 --podzielnik
    when @GIDTyp IN (7680,7681,7682,7683)
      then 20 --bilansotwarcia
    when @GIDTyp IN (2768,7686,7687,7688,7689,6272,6400,6401)
      then 21 --podnag
    when @GIDTyp IN (944)
      then 22 --prckarty
    when @GIDTyp IN (32)
      then 23 --kntkarty
    when @GIDTyp IN (368)
      then 24 --srtkarty
    when @GIDTyp IN (16)
      then 25 --twrkarty
    when @GIDTyp IN (768)
      then 26 --operacje
    when @GIDTyp IN (4128)
      then 27 --memdoknag
   when @GIDTyp IN (435)
      then 34 --RozniceKursowe
   when @GIDTyp IN (14448)
      then 39 --BSTNag
    else
      1 --tranag
    end)

  return @retval;
end
