/* ProdCzynnosci_IdZlecenia */

CREATE TRIGGER CDN.ProdCzynnosci_IdZlecenia
  ON CDN.ProdCzynnosci
  AFTER INSERT
AS
	SET NOCOUNT ON
	UPDATE CDN.ProdCzynnosci SET PCZ_PZLId = PZLId
	FROM (SELECT PCZ_Id Id,PPC_Zlecenie PZLId FROM inserted i JOIN CDN.ProdProcesy ON PPC_Id = I.PCZ_Proces) Z WHERE Z.Id = PCZ_Id
	
	SET NOCOUNT OFF
