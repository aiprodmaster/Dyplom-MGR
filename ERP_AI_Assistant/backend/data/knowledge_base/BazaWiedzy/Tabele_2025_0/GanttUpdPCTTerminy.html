/* GanttUpdPCTTerminy */

CREATE PROCEDURE CDN.GanttUpdPCTTerminy 
	@pctTerminOd    INT, 
	@pctTerminDo    INT, 
	@pctId          INT 
 
AS 
 
update CDN.ProdCzynnosciTerminy 
set PCT_CzasRozliczeniowy = (1.0*(@pctTerminDo-@pctTerminOd)/case when (PCT_TerminDo - PCT_TerminOd)= 0 then 1 else (PCT_TerminDo - PCT_TerminOd) end  ) * PCT_CzasRozliczeniowy,
PCT_TerminOd = @pctTerminOd, PCT_TerminDo = @pctTerminDo
where PCT_ID = 	@pctId
