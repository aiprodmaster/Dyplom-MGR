/* Prod_TechnologiaCzynnoscIdent */

CREATE FUNCTION CDN.Prod_TechnologiaCzynnoscIdent (@Id INT)
RETURNS VARCHAR(400)
AS
BEGIN
declare @nameDzl as varchar(400)
declare @namePom as varchar(40)
declare @IdParent as Int
SET @nameDzl =  ''
SELECT @IdParent = PTC_Ojciec from CDN.ProdTechnologiaCzynnosci
where PTC_Id = @Id
WHILE @Id > 0
BEGIN
 select @namePom =  Cast(PTC_Id As Varchar) , @Id = PTC_Ojciec from CDN.ProdTechnologiaCzynnosci where PTC_Id = @Id
 SET @nameDzl =  @namePom + '.'+ @nameDzl
END
IF Len(@nameDzl) > 0
   SET @nameDzl = Substring(@nameDzl, 1, len(@nameDzl)-1)
Return @nameDzl
END
