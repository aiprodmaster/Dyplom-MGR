/* PolaczenieInfo */

CREATE FUNCTION CDN.PolaczenieInfo(@param int)
  --@param : Jeżeli	  1 - nr_wątku
					--2 - SPID wątku głównego
					--3 - Identyfikator sesji
					--4 - Akronim operatora
					--5 - GIDNumer operatora
  RETURNS VARCHAR(100)
  AS
  BEGIN

	RETURN (SELECT StrValue FROM CDN.Split(Program_Name(),':',@param +1))
	  
  END
