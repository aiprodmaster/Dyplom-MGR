/* [PlanySkladnikiObciazenia_Delete] */

CREATE TRIGGER [CDN].[PlanySkladnikiObciazenia_Delete]
  ON [CDN].[PlanySkladnikiObciazenia]
  FOR DELETE
AS
SET NOCOUNT ON

    -- obsługa kaskadowego kasowania wierszy z tabeli CDN.PlanyPozycjeSkladniki
    DELETE CDN.PlanyPozycjeSkladniki
    FROM CDN.PlanyPozycjeSkladniki, deleted
    WHERE
          deleted.PSO_Id = CDN.PlanyPozycjeSkladniki.PPS_PSOId

SET NOCOUNT OFF
