/* UsunZnakiNiealfanumeryczneAI */

create function CDN.UsunZnakiNiealfanumeryczneAI(@Tekst varchar(8000))
returns varchar(8000)
as
begin
	declare @TekstBezZnakowSpecjalnych varchar(8000) = @Tekst COLLATE SQL_Latin1_General_CP1253_CI_AI;

	Declare @KeepValues as varchar(50)
    Set @KeepValues = '%[^a-z0-9]%'
    While PatIndex(@KeepValues, @TekstBezZnakowSpecjalnych) > 0
        Set @TekstBezZnakowSpecjalnych = Stuff(@TekstBezZnakowSpecjalnych, PatIndex(@KeepValues, @TekstBezZnakowSpecjalnych), 1, '')

	return @TekstBezZnakowSpecjalnych;
end;
