/* [WszystkoPL_UsunParametrZKategorii] */

CREATE  PROCEDURE [CDN].[WszystkoPL_UsunParametrZKategorii]
@WszystkoPLKpkId  INT, -- ID slownika z wszystko.pl
@WszystkoPLCategoryId INT
AS
SET NOCOUNT ON

DECLARE @KpkId INT

SET @KpkId = (SELECT KPK_Id FROM cdn.KgtParametryKategorii WHERE KPK_WszystkoPLId = @WszystkoPLKpkId AND KPK_KategoriaId = @WszystkoPLCategoryId);

BEGIN
DELETE FROM cdn.KgtWartoscParametrDlaTowaru WHERE KWPT_ParametrKategoriiId = @KpkId
DELETE FROM cdn.KgtParametryKategorii WHERE Kpk_Id = @KpkId
END
