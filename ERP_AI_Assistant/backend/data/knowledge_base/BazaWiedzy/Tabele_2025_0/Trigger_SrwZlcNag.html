/* SrwZlcNag_LastMod */

CREATE TRIGGER CDN.SrwZlcNag_LastMod
ON CDN.SrwZlcNag
FOR UPDATE
AS
BEGIN

  IF UPDATE(SZN_ZewnetrznySys) OR UPDATE(SZN_ZewnetrznyID)
  BEGIN
	UPDATE CDN.SrwZlcNag
	SET SZN_LastMod = datediff(s, convert(DATETIME, '1990-01-01', 120), getdate())
	FROM CDN.SrwZlcNag T
	JOIN inserted I ON T.SZN_ID = I.SZN_ID
  END
END
