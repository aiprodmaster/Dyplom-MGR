/* [TwrLastModC] */

CREATE TRIGGER [TwrLastModC]
ON [CDN].[TwrKarty]
FOR INSERT, UPDATE
AS
BEGIN
	IF NOT UPDATE(Twr_Oddzialowy)
		BEGIN
			UPDATE T
				SET T.Twr_LastModC = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
			FROM CDN.TwrKarty T, inserted
			WHERE T.Twr_GIDNumer = inserted.Twr_GIDNumer
		END
END
