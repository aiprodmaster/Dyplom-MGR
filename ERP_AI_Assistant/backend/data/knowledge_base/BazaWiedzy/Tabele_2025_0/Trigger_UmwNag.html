/* [UmwNag_DeleteAneks] */

CREATE TRIGGER [CDN].[UmwNag_DeleteAneks]
  ON [CDN].[UmwNag]
  FOR DELETE
AS
BEGIN
SET NOCOUNT ON
        if exists(select * FROM deleted inner join CDN.UmwNag umn on umn.UmN_Id = deleted.UmN_AneksId where umn.UmN_UmowaId = 0 )
            UPDATE CDN.UmwNag SET UmN_AneksId = 0
            FROM deleted inner join CDN.UmwNag umn on umn.UmN_Id = deleted.UmN_AneksId 
        else
			UPDATE CDN.UmwNag SET UmN_AneksId = deleted.UmN_AneksId
			FROM deleted inner join CDN.UmwNag umn on umn.UmN_Id = deleted.UmN_UmowaId

SET NOCOUNT OFF
END
