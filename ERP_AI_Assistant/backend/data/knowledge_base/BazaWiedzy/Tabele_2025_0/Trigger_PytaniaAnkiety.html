/* PytaniaAnkiety_DeletePrzenumeruj */

CREATE TRIGGER CDN.PytaniaAnkiety_DeletePrzenumeruj
   ON  CDN.PytaniaAnkiety
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.PytaniaAnkiety SET PAN_Pozycja = E.PAN_Pozycja - 1
FROM CDN.PytaniaAnkiety E
INNER JOIN Deleted D ON E.PAN_AntId = D.PAN_AntId
WHERE D.PAN_Pozycja < E.PAN_Pozycja


SET NOCOUNT OFF
END
