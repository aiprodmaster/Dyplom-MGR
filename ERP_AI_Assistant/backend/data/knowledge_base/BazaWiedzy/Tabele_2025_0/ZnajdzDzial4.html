/* [ZnajdzDzial4] */

CREATE PROCEDURE [CDN].[ZnajdzDzial4]
@FRSGRO_Typ INT,
@FRSGRO_Lp INT,
@FRSGRO_Firma INT,
@sDzlKod VARCHAR(40)

AS SET NOCOUNT ON

select * from CDN.frmStruktura where frs_grotyp = @FRSGRO_Typ AND frs_grofirma=@FRSGRO_Firma AND frs_grolp = @FRSGRO_Lp AND
     frs_nazwa = ''+@sDzlKod+''
