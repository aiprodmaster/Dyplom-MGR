/* [Wysylki_LastModTStamp] */

create trigger [CDN].[Wysylki_LastModTStamp]
ON [CDN].[Wysylki]
For update
As
SET NOCOUNT ON
  UPDATE CDN.Wysylki SET WYS_LastMod=DateDiff(second,convert(datetime,'1990-01-01',120),getdate())
  FROM CDN.Wysylki Z JOIN deleted I ON Z.WYS_GIDNumer=I.WYS_GIDNumer
  where Z.WYS_NumerObcy <> I.WYS_NumerObcy
SET NOCOUNT OFF
