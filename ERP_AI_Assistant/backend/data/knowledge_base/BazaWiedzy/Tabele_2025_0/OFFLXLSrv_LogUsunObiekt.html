/* OFFLXLSrv_LogUsunObiekt */

CREATE PROCEDURE CDN.OFFLXLSrv_LogUsunObiekt
@PcKID		INT,
@ObiTyp		SMALLINT,
@ObiNumer	INT,
@ObiektTyp	INT,
@ObiektID	INT
AS

	DECLARE @TS INT
	SET @TS = DATEDIFF(SS,CONVERT(DATETIME,'1990-01-01',120),CONVERT(DATETIME,GETDATE(),120))

	INSERT INTO CDN.PicoUsuniete
	(
		PcU_PcKId
		,PcU_ObiTyp
		,PcU_ObiNumer
		,PcU_ObiektTyp
		,PcU_ObiektID		
		,PcU_TSUsuniecia
		,PcU_Stan
	)
	VALUES
	(
		@PcKID
		,@ObiTyp
		,@ObiNumer
		,@ObiektTyp
		,@ObiektID
		,@TS
		,0
	)

	IF @@ERROR <> 0 
		RETURN 1

	RETURN 0
