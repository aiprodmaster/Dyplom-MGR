/* POS_AktualizacjaOstatniaModyfikacjaPOS */

CREATE PROCEDURE CDN.POS_AktualizacjaOstatniaModyfikacjaPOS
	@GIDNumer INT,
	@PointOfSaleId INT
AS
BEGIN
SET NOCOUNT ON;
SET CONTEXT_INFO 0x15ff2808;
UPDATE CDN.ZamNag SET ZaN_OstatniaModyfikacjaPOS = ZaN_LastMod + 1,ZaN_PsAID = @PointOfSaleId WHERE ZaN_GIDNumer = @GIDNumer;
SET NOCOUNT OFF;	
END
