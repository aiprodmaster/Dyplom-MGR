/* [Kon_CzasModyfikacji] */

CREATE TRIGGER [CDN].[Kon_CzasModyfikacji]
ON [CDN].[Konfig]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE K SET K.Kon_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.Konfig K join inserted I on K.Kon_Numer = I.Kon_Numer and K.Kon_Lp = I.Kon_Lp

	SET NOCOUNT OFF
END
