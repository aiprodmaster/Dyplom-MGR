/* [CRMWizytyNag_CzasModyfikacjiC] */

CREATE TRIGGER [CDN].[CRMWizytyNag_CzasModyfikacjiC]
   ON  [CDN].[CRMWizytyNag]
   FOR INSERT, UPDATE
AS 
BEGIN
SET NOCOUNT ON;

	UPDATE E
	SET E.CWN_CzasModyfikacjiC = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.CRMWizytyNag E, Inserted
	WHERE E.CWN_ID = Inserted.CWN_ID      

SET NOCOUNT OFF
END
