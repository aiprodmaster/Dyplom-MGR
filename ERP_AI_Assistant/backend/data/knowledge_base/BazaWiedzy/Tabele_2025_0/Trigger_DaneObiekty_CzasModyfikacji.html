/* [DaneObiekty_CzasModyfikacji] */

CREATE TRIGGER [CDN].[DaneObiekty_CzasModyfikacji]
ON [CDN].[DaneObiekty]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.DaO_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.DaneObiekty N join inserted I on N.DAO_ZalID = I.DAO_ZalID

	SET NOCOUNT OFF
END
