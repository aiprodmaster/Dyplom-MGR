/* ESKLEP_PobierzIdFormyKontaktu */

CREATE FUNCTION CDN.ESKLEP_PobierzIdFormyKontaktu (@FormaKontaktu VARCHAR(256))
RETURNS VARCHAR(36)
AS
BEGIN
	DECLARE @ID VARCHAR(36)
	SELECT @ID = CASE 
		WHEN @FormaKontaktu = 'E-mail' THEN 'C56924B9-8560-6099-677A-346A4B0430B4'
		WHEN @FormaKontaktu = 'Telefon' THEN '8AE09577-576B-1385-30CA-99852ACE6173'
		WHEN @FormaKontaktu = 'Sms' THEN '62120227-59AC-41F4-B952-2859604131F5'
		WHEN @FormaKontaktu = 'Przesyłka' THEN '68A0BB81-2350-42E8-8985-BC3F72C9E510'
	END
	RETURN @ID
END
