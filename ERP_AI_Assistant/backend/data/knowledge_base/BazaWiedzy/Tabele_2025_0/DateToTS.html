/* DateToTS */

CREATE FUNCTION CDN.DateToTS (@dt DATETIME = NULL)
RETURNS INT
AS
BEGIN
	RETURN DATEDIFF(ss, '1990-01-01', coalesce(@dt, getdate()))
END

        

          /* DateToClarion */

CREATE FUNCTION CDN.DateToClarion (@dt DATETIME = NULL)
RETURNS INT
AS
BEGIN
	RETURN DATEDIFF(dd, '1800-12-28', coalesce(@dt, getdate()))
END
