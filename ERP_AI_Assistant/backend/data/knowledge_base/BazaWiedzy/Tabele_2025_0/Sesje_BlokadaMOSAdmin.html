/* Sesje_BlokadaMOSAdmin */

CREATE TRIGGER CDN.Sesje_BlokadaMOSAdmin ON CDN.Sesje
 FOR DELETE
AS
SET NOCOUNT ON
    -- blokada usunięcia operatora MOSAdmin
    IF EXISTS
       ( SELECT * FROM  deleted
          WHERE
                deleted.[SES_SesjaZew] = 2
       )
    BEGIN
      RAISERROR('Nie można usunąć sesji WMS [Sesje_BlokadaMOSAdmin]. ', 16, 1)
      ROLLBACK TRAN
      SET NOCOUNT OFF
      RETURN
    END
