/* Aktualizacja pola Knt_KSeFWyslij w tabeli CDN.KntKarty */

UPDATE CDN.KntKarty
SET Knt_KSeFWyslij = 0 
WHERE Knt_WSTO = 1 OR Knt_Status = 2

        

          /* Aktualizacja pola KnW_KSeFWyslij w tabeli CDN.KntWzorce */

UPDATE CDN.KntWzorce
SET KnW_KSeFWyslij = 0 
WHERE KnW_WSTO = 1 OR KnW_Status = 2

        

          /* Aktualizacja pola DZM_WartoscParametru w tabeli CDN.DtsZmienne */

UPDATE cdn.DtsZmienne 
SET DZM_WartoscParametru= '' 
FROM cdn.DtsZestawy
WHERE DZM_DZEID = DZE_ID AND (DZM_NazwaParametru = 'SourceUID' 
OR DZM_NazwaParametru = 'SourcePWD' 
OR DZM_NazwaParametru = 'DestUID' 
OR DZM_NazwaParametru = 'DestPWD') AND DZE_ZKonfiguracji = 1
