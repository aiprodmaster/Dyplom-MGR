/* [WSZYSTKOPL_AktualizujToken] */

CREATE  PROCEDURE [CDN].[WSZYSTKOPL_AktualizujToken]
@AccessToken  VARCHAR(2048), --   	
@RefreshToken  VARCHAR(2048) --   
AS
SET NOCOUNT ON


UPDATE CDN.Picokonfig SET PcK_WszystkoPLAccessToken= @AccessToken , PcK_WszystkoPLRefreshToken = @RefreshToken where PcK_Typ = 9

SET NOCOUNT OFF
