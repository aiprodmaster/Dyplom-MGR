/* Raporty_CzasModyfikacji_OFFL */

CREATE TRIGGER CDN.Raporty_CzasModyfikacji_OFFL
ON CDN.Raporty
FOR UPDATE
AS
BEGIN
  IF UPDATE(KRP_ZewnetrznyId)
  BEGIN
    UPDATE CDN.Raporty SET KRP_CzasModyfikacji = datediff(s,convert(datetime,'1990-01-01',120),getdate())
    FROM CDN.Raporty R JOIN inserted I ON R.KRP_GIDNumer = I.KRP_GIDNumer 
  END
END
