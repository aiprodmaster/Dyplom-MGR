/* RetroKontrahenci_DeleteRKO */

CREATE TRIGGER CDN.RetroKontrahenci_DeleteRKO
  ON CDN.RetroKontrahenci
  FOR DELETE
  AS
SET NOCOUNT ON

  DELETE CDN.RetroKntOkresy
  FROM deleted 
  join CDN.RetroKntOkresy ON deleted.RKt_KntTyp=CDN.RetroKntOkresy.RKO_KntTyp and deleted.RKt_KntFirma=CDN.RetroKntOkresy.RKO_KntFirma and
	deleted.RKt_KntNumer=CDN.RetroKntOkresy.RKO_KntNumer and deleted.RKt_KntLp=CDN.RetroKntOkresy.RKO_KntLp and  deleted.RKt_RtKId=CDN.RetroKntOkresy.RKO_RtKId


SET NOCOUNT OFF
