/* CDN.SamPrzejazdy */

CREATE TABLE CDN.SamPrzejazdy
  (
      SaP_Id                          INTEGER        IDENTITY , --- LONG
      SaP_DataWyjazdu                 INTEGER        , --- LONG
      SaP_DataPrzyjazdu               INTEGER        , --- LONG
      SaP_Odleglosc                   DECIMAL ( 7,2) , --- DECIMAL(7,2)
      SaP_SamId                       INTEGER        , --- LONG
      SaP_KierTyp                     SMALLINT       , --- SHORT
      SaP_KierNumer                   INTEGER        , --- LONG
      SaP_RodzajPrzejazdu             INTEGER        , --- LONG
      SaP_OpeNumer                    INTEGER        , --- LONG
      SaP_DataZapisu                  INTEGER        , --- LONG
      CONSTRAINT SaP_Primary PRIMARY KEY
        (
                          SaP_Id
        )
  )
END;

;
