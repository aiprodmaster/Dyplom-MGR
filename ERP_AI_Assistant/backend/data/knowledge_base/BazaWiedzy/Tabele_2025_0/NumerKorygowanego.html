/* [NumerKorygowanego] */

CREATE FUNCTION [CDN].[NumerKorygowanego] (@gidtyp integer,@gidnumer integer)
RETURNS integer AS
BEGIN
	DECLARE @sav_gidtyp integer;
	DECLARE @sav_gidnumer integer;
	SET @sav_gidtyp=0
	SET @sav_gidnumer=0
	WHILE @gidtyp&8 = 8 AND (@sav_gidtyp<>@gidtyp OR @sav_gidnumer<>@gidnumer)
	BEGIN
		SET @sav_gidtyp=@gidtyp
		SET @sav_gidnumer=@gidnumer
		SELECT @gidnumer=B.TrN_GIDNumer,@gidtyp=B.TrN_GIDTyp FROM CDN.TraNag AS A JOIN CDN.Tranag AS B ON B.TrN_GIDTyp=A.TrN_ZwRTyp AND B.TrN_GIDNumer=A.TrN_ZwRNumer AND A.TrN_GIDTyp=@gidtyp AND A.TrN_GIDnumer=@gidnumer
	END
	RETURN @gidnumer
END
