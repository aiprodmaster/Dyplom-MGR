/* [PobierzDrzewoGrupTowarowych] */

CREATE FUNCTION [CDN].[PobierzDrzewoGrupTowarowych] (
	@GIDNumer INT
	, @GIDTyp SMALLINT = - 16
	, @GrONumer INT = NULL
	, @SciezkaTyp TINYINT = NULL
	, @Separator VARCHAR(3) = '\'
	, @Kierunek TINYINT = 0
	)
RETURNS TABLE
AS
RETURN (
		WITH DrzewoGrupTwr AS (
				SELECT TwG_Kod Kod
					, TwG_Nazwa Nazwa
					, TwG_GIDNumer GIDNumer
					, TwG_GIDTyp GIDTyp
					, TwG_GrONumer GrONumer
					, 1 Poziom
					, CASE @SciezkaTyp
						WHEN 1
							THEN CONVERT(VARCHAR(1000), TwG_Kod)
						WHEN 2
							THEN CONVERT(VARCHAR(1000), TwG_Nazwa)
						WHEN 3
							THEN CONVERT(VARCHAR(1000), TwG_GIDNumer)
						ELSE ''
						END Sciezka
				FROM CDN.TwrGrupy
				WHERE TwG_GIDNumer = @GIDNumer
					AND TwG_GIDTyp = ISNULL(@GIDTyp, - 16)
					AND TwG_GrONumer = ISNULL(@GrONumer, TwG_GrONumer)
				
				UNION ALL
				
				SELECT TwG_Kod Kod
					, TwG_Nazwa Nazwa
					, TwG_GIDNumer GIDNumer
					, TwG_GIDTyp GIDTyp
					, TwG_GrONumer GrONumer
					, Poziom + 1 Poziom
					, CASE @SciezkaTyp
						WHEN 1
							THEN CONVERT(VARCHAR(1000), (
										CASE 
											WHEN @Kierunek = 0
												THEN Sciezka
											ELSE TwG_Kod
											END + @Separator + CASE 
											WHEN @Kierunek = 0
												THEN TwG_Kod
											ELSE Sciezka
											END
										))
						WHEN 2
							THEN CONVERT(VARCHAR(1000), (
										CASE 
											WHEN @Kierunek = 0
												THEN Sciezka
											ELSE TwG_Nazwa
											END + @Separator + CASE 
											WHEN @Kierunek = 0
												THEN TwG_Nazwa
											ELSE Sciezka
											END
										))
						WHEN 3
							THEN CONVERT(VARCHAR(1000), (
										CASE 
											WHEN @Kierunek = 0
												THEN Sciezka
											ELSE CONVERT(VARCHAR(1000), TwG_GIDNumer)
											END + @Separator + CASE 
											WHEN @Kierunek = 0
												THEN CONVERT(VARCHAR(1000), TwG_GIDNumer)
											ELSE Sciezka
											END
										))
						ELSE ''
						END Sciezka
				FROM DrzewoGrupTwr
				INNER JOIN CDN.TwrGrupy
					ON (
							GIDNumer = CASE 
								WHEN @Kierunek = 0
									THEN TwG_GrONumer
								ELSE NULL
								END
							OR GrONumer = CASE 
								WHEN @Kierunek = 0
									THEN NULL
								ELSE TwG_GIDNumer
								END
							)
						AND TwG_GIDTyp = - 16
				)
		SELECT *
		FROM DrzewoGrupTwr
		)
