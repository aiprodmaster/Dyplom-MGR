/* GIDFirma */

CREATE FUNCTION CDN.GIDFirma()
RETURNS INT
AS
BEGIN
DECLARE @Sys VARCHAR(20)
DECLARE	@F1	varchar(10)
DECLARE	@F2	varchar(10)
DECLARE	@F3	varchar(10)

SELECT @Sys=SYS_Wartosc FROM CDN.systemcdn WHERE SYS_ID=2

set @F1  = SUBSTRING(@Sys,0,PATINDEX('%.%',@Sys))
set @Sys = SUBSTRING(@Sys,PATINDEX('%.%',@Sys)+1,LEN(@Sys))
set @F2  = SUBSTRING(@Sys,0,PATINDEX('%.%',@Sys))
set @Sys = SUBSTRING(@Sys,PATINDEX('%.%',@Sys)+1,LEN(@Sys))
set @F3  = SUBSTRING(@Sys,0,PATINDEX('%.%',@Sys))
set @Sys = SUBSTRING(@Sys,PATINDEX('%.%',@Sys)+1,LEN(@Sys))
RETURN  CAST(@Sys as int)+CAST(@F3 as int)*POWER(2,8)+CAST(@F2 as int)*POWER(2,16)
END
