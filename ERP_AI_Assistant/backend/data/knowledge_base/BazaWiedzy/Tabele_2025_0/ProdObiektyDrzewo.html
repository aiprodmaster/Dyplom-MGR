/* ProdObiektyDrzewo */

CREATE FUNCTION CDN.ProdObiektyDrzewo (@Obiekt int, @Kierunek smallint = 1)
RETURNS @Obiekty TABLE (
  Id int,
  Kod varchar(40),
  Nazwa varchar(100),
  Poziom int,
  Ojciec int
)
AS
BEGIN
	DECLARE @Poziom INT
	SET @Poziom = 0

	INSERT INTO @Obiekty
    SELECT POB_Id, POB_Kod, POB_Nazwa, 0, POB_Ojciec
    FROM CDN.ProdObiekty
    WHERE POB_Id = @Obiekt or (@Obiekt=0 and POB_Ojciec=0)

	IF @Kierunek < 0
	BEGIN
		WHILE 1 = 1
		BEGIN
			INSERT INTO @Obiekty
			SELECT POB_Id, POB_Kod, POB_Nazwa, Poziom + @Kierunek, POB_Ojciec
			FROM @Obiekty INNER JOIN CDN.ProdObiekty
				ON POB_Id = Ojciec
			WHERE Poziom = @Poziom
		  
			IF @@ROWCOUNT = 0 
				BREAK
			SET @Poziom = @Poziom + @Kierunek
		END
	END
	ELSE
	BEGIN
		WHILE 1 = 1
		BEGIN
			INSERT INTO @Obiekty
			SELECT POB_Id, POB_Kod, POB_Nazwa, Poziom + @Kierunek, POB_Ojciec
			FROM @Obiekty INNER JOIN CDN.ProdObiekty
				ON POB_Ojciec = Id
			WHERE Poziom = @Poziom
		  
			IF @@ROWCOUNT = 0 
				BREAK
			SET @Poziom = @Poziom + @Kierunek
		END
	END
	
	RETURN
END
