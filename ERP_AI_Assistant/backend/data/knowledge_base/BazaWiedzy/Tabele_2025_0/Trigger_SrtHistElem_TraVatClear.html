/* SrtHistElem_Delete_TraVatClear */

CREATE TRIGGER CDN.SrtHistElem_Delete_TraVatClear
  ON CDN.SrtHistElem
  AFTER DELETE
AS
SET NOCOUNT ON
    -- czyszczenie powiązania CDN.TraVat z CDN.SrtHistElem    
    UPDATE CDN.TraVat SET TrV_SHENumer = 0, TrV_SHELp = 0 
    FROM CDN.TraVat tv INNER JOIN deleted d ON tv.TrV_SheNumer = d.She_GIDNumer AND tv.TrV_SheLp = d.She_GIDLp    

SET NOCOUNT OFF
