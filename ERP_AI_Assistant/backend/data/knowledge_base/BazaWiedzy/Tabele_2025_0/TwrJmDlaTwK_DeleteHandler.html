/* TwrJmDlaTwK_DeleteHandler */

CREATE TRIGGER CDN.TwrJmDlaTwK_DeleteHandler
  ON CDN.TwrJm
  FOR DELETE
AS
SET NOCOUNT ON

    -- obsługa kaskadowego kasowania wierszy z tabeli CDN.TwrKody
    DELETE CDN.TwrKody
    FROM CDN.TwrKody, deleted
    WHERE
          deleted.TwJ_TwrNumer = CDN.TwrKody.TwK_TwrNumer
      AND deleted.TwJ_JmZ = CDN.TwrKody.TwK_Jm

SET NOCOUNT OFF
