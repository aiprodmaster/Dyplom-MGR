/* DXZapiszFormat */

CREATE PROCEDURE CDN.DXZapiszFormat
@Format NVARCHAR(MAX),
@Procedura VARCHAR(50),
@Operator INT,
@IDKontrolki INT
AS
	
UPDATE CDN.FormatyList SET FLS_Format = @Format WHERE FLS_Procedura = @Procedura AND FLS_IDOperatora = @Operator AND FLS_IDKontrolki = @IDKontrolki
IF @@ROWCOUNT=0
	INSERT INTO CDN.FormatyList (FLS_Modul,FLS_Procedura,FLS_IDOperatora,FLS_IDKontrolki,FLS_IDFormatu,FLS_NazwaFormatu,FLS_Flaga,FLS_Format,FLS_VarLine) VALUES('ERP XL',@Procedura,@Operator,@IDKontrolki,1,N'Domyślny format DX',1,@Format,'')
