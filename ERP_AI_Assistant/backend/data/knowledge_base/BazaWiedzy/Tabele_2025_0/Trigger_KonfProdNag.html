/* [KonfProdNag_DeleteZmienProdTech] */

CREATE TRIGGER [CDN].[KonfProdNag_DeleteZmienProdTech]
   ON  [CDN].[KonfProdNag]
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.ProdTechnologia SET PTE_ZrdNumer = 0, PTE_ZrdTyp = 0
FROM CDN.ProdTechnologia
INNER JOIN deleted on KPN_Id = PTE_ZrdNumer
WHERE PTE_ZrdTyp = 14371

SET NOCOUNT OFF
END
