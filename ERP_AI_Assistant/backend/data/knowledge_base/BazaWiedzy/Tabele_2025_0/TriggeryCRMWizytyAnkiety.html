/* [CRMWizytyAnkiety_Delete] */

CREATE TRIGGER [CDN].[CRMWizytyAnkiety_Delete]
  ON [CDN].[CRMWizytyAnkiety]
  FOR DELETE
AS
SET NOCOUNT ON

    -- obsługa kaskadowego kasowania wierszy z tabeli CDN.CRMDefZadania
    DELETE CDN.Zadania
    FROM CDN.Zadania, deleted
    WHERE
          deleted.CWA_ZadId = CDN.Zadania.Zad_Id

SET NOCOUNT OFF
