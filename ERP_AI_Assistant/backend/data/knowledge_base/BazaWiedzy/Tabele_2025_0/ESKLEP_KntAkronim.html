/* ESKLEP_KntAkronim */

CREATE FUNCTION CDN.ESKLEP_KntAkronim (
  @ESklepID INT             -- ID e-sklepu
 ,@KntID    INT )           -- ID kontrahenta w e-sklepie
RETURNS VARCHAR(20)
BEGIN
	DECLARE @Akronim VARCHAR(20)
	SET @Akronim = 'ESKL' + CAST(@EsklepID AS VARCHAR) + '_' + CAST(@KntID AS VARCHAR)
	RETURN @Akronim
END
