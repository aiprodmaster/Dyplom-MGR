/* TraNag_LastMod_OFFL */

CREATE TRIGGER CDN.TraNag_LastMod_OFFL
ON CDN.TraNag
FOR UPDATE
AS
BEGIN
  IF UPDATE(TrN_OddDokID) OR UPDATE(TrN_OddZakazPAFA)
  BEGIN
    UPDATE CDN.TraNag SET TrN_LastMod = datediff(s,convert(datetime,'1990-01-01',120),getdate())
    FROM CDN.TraNag T JOIN inserted I ON T.TrN_GIDNumer = I.TrN_GIDNumer 
  END
END
