/* [AtrybutyKomplety_CzasModyfikacji] */

CREATE TRIGGER [CDN].[AtrybutyKomplety_CzasModyfikacji]
ON [CDN].[AtrybutyKomplety]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.akp_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.AtrybutyKomplety N join inserted I on N.AKp_Id = I.AKp_Id

	SET NOCOUNT OFF
END
