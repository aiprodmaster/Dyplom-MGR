/* Log_SzczegolyBledu */

CREATE FUNCTION CDN.Log_SzczegolyBledu
(
  @NumerBledu INT
 ,@NazwaProcedury VARCHAR(256)
 ,@NumerWiersza INT
 ,@Komunikat VARCHAR(1024)
)
RETURNS VARCHAR(1024)
AS
BEGIN


RETURN ISNULL(@NazwaProcedury,'') +  ' (wiersz: ' + CAST(@NumerWiersza AS VARCHAR(50))+ CASE WHEN @NumerBledu IS NOT NULL THEN ') - ('+ CAST(@NumerBledu AS VARCHAR(50)) + ') '  ELSE '' END + ISNULL(@Komunikat,'')

END
