/* [PlanyPozycjeSkladniki_Delete] */

CREATE TRIGGER [CDN].[PlanyPozycjeSkladniki_Delete]
  ON [CDN].[PlanyPozycjeSkladniki]
  FOR DELETE
AS
SET NOCOUNT ON

 if trigger_nestlevel(  object_ID('cdn.PlanyPozycjeSkladniki_Delete'))>1
	 RETURN

    -- obsługa kaskadowego kasowania wierszy z tabeli CDN.PlanySkladnikiObciazenia
    DELETE CDN.PlanySkladnikiObciazenia
    FROM CDN.PlanySkladnikiObciazenia, deleted
    WHERE
          deleted.PPS_PSOId = CDN.PlanySkladnikiObciazenia.PSO_Id

SET NOCOUNT OFF
