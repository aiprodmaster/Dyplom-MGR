/* KodKrajuIntrastatNaISO */

CREATE FUNCTION CDN.KodKrajuIntrastatNaISO(@KodIN VARCHAR(2))
RETURNS VARCHAR(2) AS
BEGIN
	DECLARE @KodISO VARCHAR(2)
	SELECT @KodISO = CASE
	WHEN @KodIN = 'XS' THEN 'RS'
	WHEN @KodIN = 'XX' THEN 'SK'
	ELSE @KodIN
	END
	RETURN @KodISO
END

        

          /* KodKrajuISONaIntrastat */

CREATE FUNCTION CDN.KodKrajuISONaIntrastat(@KodISO VARCHAR(2))
RETURNS VARCHAR(2) AS
BEGIN
	DECLARE @KodIN VARCHAR(2)
	SELECT @KodIN = CASE
	WHEN @KodISO = 'RS' THEN 'XS'
	WHEN @KodISO = 'SK' THEN 'XX'
	ELSE @KodISO
	END
	RETURN @KodIN
END
