/* DokDostepneMagazynyLinkiRez */

CREATE FUNCTION CDN.DokDostepneMagazynyLinkiRez (
	@lFrsId INT,
	@lFrsIdW INT,
	@lDokTyp SMALLINT,
	@lZrdTyp SMALLINT,
	@lZrdNumer INT
	)
RETURNS TABLE
AS
RETURN

SELECT MAG_GIDTyp,
	MAG_GIDFirma,
	MAG_GIDNumer,
	MAG_GIDLp,
	MAG_Kod,
	MAG_Wewnetrzny
FROM CDN.DokDostepneMagazynyLinki(@lFrsId, @lFrsIdW, @lDokTyp)
JOIN CDN.RezMagazyny ON ReM_RezTyp = @lZrdTyp
	AND ReM_RezNumer = @lZrdNumer
	AND ReM_MagNumer = Mag_GIDNumer
