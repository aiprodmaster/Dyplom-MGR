/* ECODFSKInvoiceSummary */

create procedure CDN.ECODFSKInvoiceSummary
	@TrNGidTyp integer,
	@TrNGidNumer integer,
	@TotalTaxAmount decimal(15,2),
	@TotalGrossAmount decimal(15,2),
	@TotalNetAmount decimal(15,2),
	@TotalTaxableBasis decimal(15,2),
	@PreviousTotalTaxAmount decimal(15,2),
	@PreviousTotalGrossAmount decimal(15,2),
	@PreviousTotalNetAmount decimal(15,2),
	@PreviousTotalTaxableBasis decimal(15,2),
	@CorrectionTotalTaxAmount decimal(15,2),
	@CorrectionTotalGrossAmount decimal(15,2),
	@CorrectionTotalNetAmount decimal(15,2),
	@CorrectionTotalTaxableBasis decimal(15,2)
as
	select @TotalTaxAmount as TotalTaxAmount, 
	@TotalGrossAmount as TotalGrossAmount,
	@TotalNetAmount as TotalNetAmount,
	@TotalTaxableBasis as TotalTaxableBasis,
	@PreviousTotalTaxAmount as PreviousTotalTaxAmount, 
	@PreviousTotalGrossAmount as PreviousTotalGrossAmount,
	@PreviousTotalNetAmount as PreviousTotalNetAmount,
	@PreviousTotalTaxableBasis as PreviousTotalTaxableBasis,
	@CorrectionTotalTaxAmount as CorrectionTotalTaxAmount,
	@CorrectionTotalGrossAmount as CorrectionTotalGrossAmount,
	@CorrectionTotalNetAmount as CorrectionTotalNetAmount,
	@CorrectionTotalTaxableBasis as CorrectionTotalTaxableBasis,
	case when @CorrectionTotalGrossAmount<0 then 'minus ' + CDN.slownie(@CorrectionTotalGrossAmount, '', '') 
	else CDN.slownie(@CorrectionTotalGrossAmount, '', '') end AS GrossAmountInWords;
