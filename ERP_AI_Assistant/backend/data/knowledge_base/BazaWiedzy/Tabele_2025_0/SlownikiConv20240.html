/* Dodanie nowych kategorii do tabeli SlownikiStruktura(Typy inwentaryzacji POS) 2024.0 */

set nocount on;

	if not exists(select * from cdn.SlownikiStruktura where (cast(SLS_Nazwa as BINARY) IN (cast('Typy Inwentaryzacji POS' as BINARY), cast('Typy inwentaryzacji POS'as BINARY))) and SLS_OId = 4)
	BEGIN
		insert into cdn.slownikistruktura (SLS_OId, SLS_Lp, SLS_Nazwa, SLS_Predefiniowany, SLS_WartoscL1) values(4,218,'Typy inwentaryzacji POS',218,0);
	END

	if exists(select * from cdn.SlownikiStruktura where (cast(SLS_Nazwa as BINARY) = cast('Typy Inwentaryzacji POS' as BINARY) and SLS_OId = 4))
	BEGIN
		update cdn.SlownikiStruktura
		set
			SLS_Nazwa='Typy inwentaryzacji POS'
		where
		SLS_Predefiniowany = 218
	END

	if exists(select * from cdn.Slowniki where cast(SLW_Kategoria as BINARY) = cast('Typy Inwentaryzacji POS' as BINARY))
	BEGIN
		update cdn.Slowniki
		set SLW_Kategoria = 'Typy inwentaryzacji POS'
		where cast(SLW_Kategoria as BINARY) = cast('Typy Inwentaryzacji POS' as BINARY)
	END
set nocount off;
