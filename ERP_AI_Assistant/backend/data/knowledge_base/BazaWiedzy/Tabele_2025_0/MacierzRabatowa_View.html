/* KntUpusty */

CREATE VIEW CDN.KntUpusty
AS
SELECT KPR_KntTyp AS KnU_KntTyp
	, KPR_KntFirma AS KnU_KntFirma
	, KPR_KntNumer AS KnU_KntNumer
	, TPR_LP AS KnU_KntLp --!!!
	, TPR_TwrTyp AS KnU_TwgTyp
	, TPR_TwrFirma AS KnU_TwgFirma
	, TPR_TwrNumer AS KnU_TwgNumer
	, TPR_TwrLp AS KnU_TwgLp
	, TPR_Wartosc AS KnU_Upust
	, cast('' as varchar(254)) AS KnU_UpustWzor
	, TPR_Waluta AS KnU_Symbol
	, cast(0 as smallint) AS KnU_NrKursu
	, TPR_FlagaNb AS KnU_FlagaNB
	, TPR_Notatki AS KnU_Notatki
	, TPR_ID AS KnU_TPRID
	, TPR_Typ AS KnU_Typ
FROM cdn.TwrPromocje
INNER JOIN Cdn.PrmKarty
	ON TPR_PrmId = Prm_Id
INNER JOIN cdn.KntPromocje
	ON KPR_PrmId = PRM_Id
WHERE PRM_Typ = 4

        

          /* TwrUpusty */

CREATE VIEW CDN.TwrUpusty
AS
SELECT TPR_TwrTyp AS TwU_TwrTyp
	, TPR_TwrFirma AS TwU_TwrFirma
	, TPR_TwrNumer AS TwU_TwrNumer
	, TPR_TwrLp AS TwU_TwrLp
	, TPR_Prog AS TwU_PRog
	, TPR_Wartosc AS TwU_Upust
FROM cdn.TwrPromocje
INNER JOIN Cdn.PrmKarty
	ON TPR_PrmId = PRM_Id
WHERE PRM_Typ = 3
