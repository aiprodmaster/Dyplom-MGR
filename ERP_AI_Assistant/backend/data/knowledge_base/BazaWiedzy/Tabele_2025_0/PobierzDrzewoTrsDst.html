/* [PobierzDrzewoTrsDstKuKorzeniowi] */

CREATE FUNCTION [CDN].[PobierzDrzewoTrsDstKuKorzeniowi] (@DstNumer INT)
RETURNS @DrzewoTrsDstKuKorzeniowi TABLE (
	DstNumer INT
	, TrnTStamp INT
	, GIDTyp SMALLINT
	, GIDNumer INT
	, GIDLp SMALLINT
	, SubGIDLp SMALLINT
	, DsFNumer INT
	, Poziom INT
	)
AS
BEGIN
	DECLARE @poziom INT

	SET @poziom = 0

	INSERT INTO @DrzewoTrsDstKuKorzeniowi
	SELECT TrS_DstNumer
		, TrS_TrnTStamp
		, TrS_GIDTyp
		, TrS_GIDNumer
		, TrS_GIDLp
		, TrS_SubGIDLp
		, TrS_DsFNumer
		, @poziom
	FROM cdn.TraSElem
	WHERE @DstNumer = TrS_DstNumer

	WHILE 1 = 1
	BEGIN
		SET @poziom = @poziom + 1

		INSERT INTO @DrzewoTrsDstKuKorzeniowi
		SELECT TrS_DstNumer
			, TrS_TrnTStamp
			, TrS_GIDTyp
			, TrS_GIDNumer
			, TrS_GIDLp
			, TrS_SubGIDLp
			, TrS_DsFNumer
			, @poziom
		FROM cdn.TraSElem
		WHERE TrS_DsFNumer > 0 AND TrS_DsFNumer IN (
				SELECT a.DstNumer
				FROM @DrzewoTrsDstKuKorzeniowi a
				WHERE a.Poziom = (
						SELECT MAX(b.Poziom)
						FROM @DrzewoTrsDstKuKorzeniowi b
						)
				)

		IF @@ROWCOUNT = 0
			OR @@ERROR > 0
			OR @poziom > 100 -- dodatkowe zabezpieczenie?
			BREAK
	END

	RETURN
END

	, SubGIDLp SMALLINT

	, Poziom INT
	)

BEGIN
	DECLARE @poziom INT

	SELECT TrS_DstNumer
		, TrS_TrnTStamp
		, TrS_GIDTyp
		, TrS_GIDNumer
		, TrS_GIDLp
		, TrS_SubGIDLp
		, TrS_DsFNumer
		, @poziom

	WHILE 1 = 1
	BEGIN
