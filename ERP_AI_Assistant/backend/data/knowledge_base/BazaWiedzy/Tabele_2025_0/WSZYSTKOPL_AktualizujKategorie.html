/* [WSZYSTKOPL_AktualizujKategorie] */

CREATE  PROCEDURE [CDN].[WSZYSTKOPL_AktualizujKategorie]
@WszystkoPLId  VARCHAR(2048), -- ID z wszystko.pl
@Nazwa VARCHAR(100), -- Nazwa kategorii   	
@OjciecId    INT, -- ID rodzica z wszystko.pl
@Lisc TINYINT, -- Czy jest liĹ›ciem w drzewie kategorii
@Sciezka VARCHAR(1000), -- Sciezka dla kategorii
@Poziom TINYINT -- Poziom kategorii
AS
SET NOCOUNT ON

UPDATE CDN.KatGrupyTwr SET KGT_Nazwa = @Nazwa, KGT_OjciecId = @OjciecId, KGT_Lisc = @Lisc, KGT_Sciezka = @Sciezka, KGT_Poziom = @Poziom WHERE KGT_WszystkoPlId = @WszystkoPLId
