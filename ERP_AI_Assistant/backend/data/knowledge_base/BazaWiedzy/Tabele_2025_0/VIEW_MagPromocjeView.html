/* MagPromocjeView */

CREATE VIEW CDN.MagPromocjeView
(
	MPRV_Id
	,MPRV_PrmId
	,MPRV_MagTyp
	,MPRV_MagNumer
	,MPRV_MagKod
	,MPRV_MagNazwa
)
AS 
SELECT 
	MP.MPR_Id
	,MP.MPR_PrmId
	,MP.MPR_MagTyp
	,MP.MPR_MagNumer
	,ISNULL(M.MAG_Kod, '')
	,ISNULL(M.MAG_Nazwa, '')
FROM CDN.MagPromocje MP
	LEFT OUTER JOIN CDN.Magazyny M 
		ON MP.MPR_MagNumer = M.MAG_GIDNumer AND MP.MPR_MagTyp = M.MAG_GIDTyp
