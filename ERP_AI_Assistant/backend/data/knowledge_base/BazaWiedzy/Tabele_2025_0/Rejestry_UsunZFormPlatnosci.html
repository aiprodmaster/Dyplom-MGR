/* Rejestry_UsunZFormPlatnosci */

CREATE TRIGGER CDN.Rejestry_UsunZFormPlatnosci
  ON CDN.Rejestry
  FOR DELETE
AS
SET NOCOUNT ON
    UPDATE CDN.FrmObiekty
      set FrO_GID2Numer = 0
    FROM CDN.FrmObiekty, deleted
    WHERE deleted.KAR_GIDNumer = CDN.FrmObiekty.FRO_GID2Numer
      AND CDN.FrmObiekty.Fro_GIDTyp=736 AND CDN.FrmObiekty.Fro_GID2Typ=752
SET NOCOUNT OFF
