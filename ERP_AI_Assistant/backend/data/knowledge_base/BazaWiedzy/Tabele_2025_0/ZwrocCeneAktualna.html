/* [ZwrocCeneAktualna] */

CREATE FUNCTION [CDN].[ZwrocCeneAktualna] (
	@p_TwrNumer INT
	, @p_RodzajCeny INT
	, @p_DataTS INT
	)
RETURNS TABLE
AS
RETURN (
		SELECT TOP 1 TwC_TwrTyp
			, TwC_TwrFirma
			, TwC_TwrNumer
			, TwC_TwrLp
			, TwC_ID
			, TwC_Waluta
			, TwC_NrKursu
			, TwC_Marza
			, TwC_Zaok
			, TwC_Offset
			, TwC_RodzajPodstawy
			, TwC_Wartosc
			, TwC_Aktualizacja
			, TwC_Priorytet
			, TwC_CzasModyfikacji
			, TwC_KosztyDodatkowe
			, TwC_DokTyp
			, TwC_DokFirma
			, TwC_DokNumer
			, TwC_DokLp
			, TwC_TcnId
			, TwC_DataOd
			, TwC_KntTyp
			, TwC_KntFirma
			, TwC_KntNumer
			, TwC_KntLp
			, TwC_Aktywny
		FROM CDN.TwrCeny
		WHERE TwC_TwrNumer = @p_TwrNumer
			AND TwC_TwrLp = @p_RodzajCeny
			AND TwC_DataOd <= @p_DataTS
		ORDER BY TwC_DataOd DESC
			, TwC_CzasModyfikacji DESC
		)

        

          /* [ZwrocCeneAktualnaDostawcy] */

CREATE FUNCTION [CDN].[ZwrocCeneAktualnaDostawcy] (
	@p_TwrNumer INT
	, @p_KntNumer INT
	, @p_KntTyp SMALLINT
	--, @p_KlasaKnt INT
	, @p_Data INT
	)
RETURNS TABLE
AS
RETURN (
		SELECT TOP 1 *
		FROM CDN.TwrDost
		INNER JOIN CDN.TwrDostCeny ON TWD_Id = TDC_TwdId
		WHERE TWD_TwrNumer = @p_TwrNumer
			AND TWD_TwrTyp = 16
			AND TWD_KntNumer = @p_KntNumer
			AND TWD_KntTyp = @p_KntTyp
			AND TWD_KlasaKnt = 8 --@p_KlasaKnt
			AND TDC_DataOd <= @p_Data
		ORDER BY TDC_DataOd DESC
			, TDC_CzasModyfikacji DESC
		)
