/* [POS_RejestrujPosAgenta] */

CREATE PROCEDURE [CDN].[POS_RejestrujPosAgenta]
	@POSAgentKod NVARCHAR(64),
	@POSAgentGUID NVARCHAR(64),
	@POSAgentHost NVARCHAR(64),
	@POSAgentPort NVARCHAR(64),
	@POSAgentProfil NVARCHAR(64)
AS
BEGIN
	update cdn.picokonfig set pck_posagentguid=@POSAgentGUID,PcK_POSAgentHost=@POSAgentHost,PcK_POSAgentPort=@POSAgentPort,Pck_POSAgentProfil=@POSAgentProfil where pck_posagentkod=@POSAgentKod
END
