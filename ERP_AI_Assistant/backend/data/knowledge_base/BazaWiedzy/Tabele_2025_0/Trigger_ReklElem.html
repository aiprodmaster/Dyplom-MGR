/* ReklElem_DeletePrzenumeruj */

CREATE TRIGGER CDN.ReklElem_DeletePrzenumeruj
   ON  CDN.ReklElem
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.ReklElem SET RlE_Pozycja = I.RlE_Pozycja - 1
FROM CDN.ReklElem I
INNER JOIN Deleted D ON I.RlE_RlNId = D.RlE_RlNId
WHERE D.RlE_Pozycja < I.RlE_Pozycja

SET NOCOUNT OFF
END
