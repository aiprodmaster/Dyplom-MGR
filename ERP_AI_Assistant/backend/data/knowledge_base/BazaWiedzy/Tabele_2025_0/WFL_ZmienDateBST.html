/* WFL_ZmienDateBST */

create procedure cdn.WFL_ZmienDateBST
  @BstId INT,
  @DataRealiz INT
AS
update  z
set WFZ_DataRealizacji = @DataRealiz,
WFZ_TerminRealizacji = @DataRealiz
from cdn.WF_Zadania z
inner join cdn.WF_Procesy p
on p.WFP_ID=z.WFZ_WFPID
join cdn.WF_DefProcesy dp
on dp.WDP_ID=p.WFP_WDPID
join cdn.WF_DefAkcje da
on dp.WDP_ID= da.WDA_WDPID

where WDA_Kluczowa = 830
and WFP_OBITyp = 14448
and WFP_OBINumer = @BstId
and WDP_Stan = 1
and WDP_Nazwa = 'Automatyczne przeliczanie bilansu stanu towarów'
