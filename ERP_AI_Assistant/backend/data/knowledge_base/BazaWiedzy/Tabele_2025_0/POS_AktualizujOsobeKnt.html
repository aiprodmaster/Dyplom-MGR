/* POS_AktualizujOsobeKnt */

CREATE PROCEDURE CDN.POS_AktualizujOsobeKnt
	@Telefon NVARCHAR(128) = NULL,
	@Mail NVARCHAR(128) = NULL,
	@KnSGUID NVARCHAR(41)
AS
BEGIN
	UPDATE CDN.KntOsoby SET KnS_Telefon = COALESCE(@Telefon,KnS_Telefon),KnS_EMail = COALESCE(@Mail,KnS_EMail) WHERE KnS_GUID = @KnSGUID
END
