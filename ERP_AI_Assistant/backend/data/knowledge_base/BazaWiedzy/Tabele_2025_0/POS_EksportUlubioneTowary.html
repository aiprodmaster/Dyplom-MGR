/* [POS_EksportUlubioneTowary] */

CREATE PROCEDURE [CDN].[POS_EksportUlubioneTowary] 
	@rowVersion bigint,
	@PointOfSaleId int
AS
BEGIN
SET NOCOUNT ON;	

	DECLARE @PckId int 
	select @PckId = PcK_PicoId from cdn.PicoStanowiska 
	join cdn.PicoKonfig on PSa_PcKID = PcK_PicoID
	where PSa_ID = @PointOfSaleId

	select 
		0 as [@RowVersion], (
		select 
			PUT_TwrNumer as [@ProductId],
			PUT_Pozycja as [@OrdinalNumber]
	
		FROM CDN.PicoUlubioneTwr 
		JOIN cdn.PicoKonfig  on PUT_PicoID = PcK_PicoID
		WHERE PUT_PicoID = @PckId

	for xml path('FavouriteProduct'), type)
	for xml path('FavouriteProducts')
	SET NOCOUNT OFF;	
END
