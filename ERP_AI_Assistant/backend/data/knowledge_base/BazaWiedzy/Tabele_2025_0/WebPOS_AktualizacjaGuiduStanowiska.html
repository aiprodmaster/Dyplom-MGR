/* WebPOS_AktualizacjaGuiduStanowiska */

CREATE PROCEDURE CDN.WebPOS_AktualizacjaGuiduStanowiska
(
	@PSaID INTEGER,			-- ID rekordu CDN.PicoStanowiska
	@NewGuid VARCHAR(41)	-- Nowy Guid
)
AS
BEGIN
	UPDATE CDN.PicoStanowiska
	SET PSa_GUID=CAST(@NewGuid as UNIQUEIDENTIFIER)
	WHERE PSa_ID = @PSaID
END
