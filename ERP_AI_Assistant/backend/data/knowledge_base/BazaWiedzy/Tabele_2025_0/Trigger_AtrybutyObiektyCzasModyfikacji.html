/* [AtrybutyObiekty_CzasModyfikacji] */

CREATE TRIGGER [CDN].[AtrybutyObiekty_CzasModyfikacji]
ON [CDN].[AtrybutyObiekty]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.ato_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.AtrybutyObiekty N join inserted I on N.AtO_GIDTyp = I.AtO_GIDTyp and N.AtO_Element = I.AtO_Element
		and N.AtO_AtKId = I.AtO_AtKId

	SET NOCOUNT OFF
END
