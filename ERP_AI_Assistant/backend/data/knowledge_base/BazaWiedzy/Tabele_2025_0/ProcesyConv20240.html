/* Konwersja parametrów akcji kluczowej Generowanie dokumentu magazynowego na definicjach procesów */

SET NOCOUNT ON;

update cdn.WF_DefAkcje set WDA_PARAMETRY = case cdn.CzyPrzychod(WDP_TypObiektu) when 1 then 'PM,AWD,WM,ZWM' else 'WM,ZWM,PM,AWD' end 
from cdn.WF_DefProcesy
join cdn.WF_DefZadania on WDZ_WDPID=WDP_ID
join cdn.WF_DefAkcje on WDA_ID = WDZ_WDAID
where WDA_KLUCZOWA = 2055 and WDP_Stan = 1

SET NOCOUNT OFF;
