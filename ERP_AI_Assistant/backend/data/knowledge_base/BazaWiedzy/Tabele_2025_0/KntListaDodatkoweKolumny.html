/* [KntListaDodatkoweKolumny] */

CREATE FUNCTION [CDN].[KntListaDodatkoweKolumny]
(
@KntGIdNumer int,
@KntGIdTyp smallint,
@Kolumny tinyint
)
RETURNS varchar(256)
AS
BEGIN
declare @ReturnValue varchar(256)

        if @Kolumny & 1 = 1
                begin
                --tutaj zamiast poniższego własny kod ustawiający wartość dla pierwszej kolumny zakończony dwukropkiem
                        set @ReturnValue = '-:'
                end
        else
			set @ReturnValue = '-:'
				
        if @Kolumny & 2 = 2
                begin
                --tutaj zamiast poniższego własny kod ustawiający wartość dla drugiej kolumny zakończony dwukropkiem
                        set @ReturnValue = @ReturnValue + '-:'
                end
        else
			set @ReturnValue = @ReturnValue + '-:'
			                
        if @Kolumny & 4 = 4
                begin
                --tutaj zamiast poniższego własny kod ustawiający wartość dla trzeciej kolumny
                        set @ReturnValue = @ReturnValue + '-'
                end
        else
			set @ReturnValue = @ReturnValue + '-'          

        return isnull(@ReturnValue,'-:-:-')
END
