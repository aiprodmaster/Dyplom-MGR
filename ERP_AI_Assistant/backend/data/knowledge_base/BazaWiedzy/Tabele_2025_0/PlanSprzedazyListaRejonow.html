/* PlanSprzedazyListaRejonowPoziom */

CREATE FUNCTION CDN.PlanSprzedazyListaRejonowPoziom(@PSRID INT)
RETURNS INT
AS
BEGIN
	DECLARE @BEZP INT
	DECLARE @Poziom INT
	DECLARE @OjciecID INT
	SET @Poziom = 0
	SET @BEZP = 1
	SET @OjciecID = @PSRID

	WHILE @OjciecID > 0 AND @BEZP < 1000
	BEGIN
		SET @BEZP = @BEZP + 1
		SET @Poziom = @Poziom+1
		SELECT @OjciecID = Rej_OjciecID FROM CDN.Rejony WHERE Rej_ID = @OjciecID
	END	
	RETURN @Poziom
END

        

          /* PlanSprzedazyListaRejonowIdent */

CREATE FUNCTION CDN.PlanSprzedazyListaRejonowIdent(@PSRID INT)
RETURNS INT
AS
BEGIN
	DECLARE @BEZP INT
	DECLARE @Poziom INT
	DECLARE @OjciecID INT
	DECLARE @REJID INT
	SET @BEZP = 1
	SET @OjciecID = @PSRID

	WHILE @OjciecID > 0 AND @BEZP < 1000
	BEGIN
		SET @BEZP = @BEZP + 1
		SELECT @OjciecID = Rej_OjciecID,@REJID = Rej_ID FROM CDN.Rejony WHERE Rej_ID = @OjciecID
	END	
	RETURN @REJID
END
