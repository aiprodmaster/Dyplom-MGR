/* [CRMRaportyDef_CzasModyfikacjiC] */

CREATE TRIGGER [CDN].[CRMRaportyDef_CzasModyfikacjiC]
   ON  [CDN].[CRMRaportyDef]
   FOR INSERT, UPDATE
AS 
BEGIN
SET NOCOUNT ON;

	UPDATE E
	SET E.CRD_CzasModyfikacjiC = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.CRMRaportyDef E, Inserted
	WHERE E.CRD_Id = Inserted.CRD_Id   

SET NOCOUNT OFF
END
