/* ProdOkresy_Delete_AktualizujMRPwTWR */

create TRIGGER CDN.ProdOkresy_Delete_AktualizujMRPwTWR
  ON CDN.ProdOkresy
  FOR DELETE
AS
SET NOCOUNT ON

update TWR
set twr_mrpid = isnull(POK_id,0)
 FROM CDN.TwrKarty twr left join CDN.ProdOkresy on pok_obityp = 16 and POK_ObiNumer = Twr_GIDNumer
 where twr_MRPid in (select POK_Id from deleted)

        SET NOCOUNT OFF
