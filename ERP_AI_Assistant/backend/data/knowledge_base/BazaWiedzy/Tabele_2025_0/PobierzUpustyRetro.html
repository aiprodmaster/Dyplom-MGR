/* [PobierzUpustyRetroDlaKnt] */

CREATE FUNCTION [CDN].[PobierzUpustyRetroDlaKnt] (@p_KntNumer INT)
RETURNS TABLE
AS
RETURN (
		WITH UpustyRetro AS (
				SELECT cast(RtK_Nazwa AS VARCHAR(201)) cPrmNazwa
					, cast('' AS VARCHAR(201)) cTwrNazwa
					, cast(RtK_Nazwa AS VARCHAR(201)) cNazwa
					, 1 lPPoziom
					, - 1 lPrmLp
					, cast(0 AS SMALLINT) lTyp
					, RtK_Id cPrmId
					, - 1 lTwrNumer
					, CASE 
						WHEN RtK_DataOd < 93890
							THEN 86400 * (RtK_DataOd - 69035)
						ELSE 2147483647
						END lPrmDataOd
					, CASE 
						WHEN RtK_DataDo < 93890
							THEN 86400 * (RtK_DataDo - 69035)
						ELSE 2147483647
						END lPrmDataDo
					, CONVERT(VARCHAR(20), isnull(NajmniejszyRetroProg.RtP_Rabat, 0)) AS cWartosc
					, '' cWartWaluta
					, RtK_RodzajCeny lRodzajCeny
					, 0 bCykliczna
					, CASE 
						WHEN RtP_TypRabatu = 1
							THEN 'Netto'
						ELSE ''
						END cNettoBrutto
					, isnull(NajmniejszyRetroProg.RtP_WartoscProgu, 0) dProgWartosc
					, 1 lRodzajDok
					, isnull(NajmniejszyRetroProg.RtP_TypRabatu, 1) lRodzajUpustu
					, 0 bPakietowa
					, - 1 lPrmTyp
					, - 1 bJestDPR
					, 0 bProgTyp
					, 0 lCyklRodzaj
					, 0 lCyklCzestotliwosc
					, 0 lCyklWystepowanie
					, 0 lCyklDniOd
					, 0 lCyklDniDo
				FROM cdn.RetroKontrahenci
				INNER JOIN cdn.RetroKarty ON RtK_Id = RKt_RtKId
				OUTER APPLY (
					SELECT TOP 1 *
					FROM cdn.RetroProgi
					WHERE RtP_RtKId = RtK_Id
					ORDER BY RtP_WartoscProgu ASC
					) NajmniejszyRetroProg --ON NajmniejszyRetroProg.RtP_RtKId = RtK_Id
				WHERE RKt_KntNumer = @p_KntNumer
					AND RKt_KntTyp = 32
					OR RKt_KntNumer IN (
						SELECT KnL_GrONumer
						FROM cdn.KntLinki
						WHERE KnL_GIDNumer = @p_KntNumer
							AND KnL_GIDTyp = 32
						)
				
				UNION ALL
				
				SELECT cast(TGV_Kod AS VARCHAR(201)) cPrmNazwa
					, cast(TGV_Nazwa AS VARCHAR(201)) cTwrNazwa
					, cNazwa
					, lPPoziom + 1 lPPoziom
					, lPrmLp
					, RtT_TwrTyp lTyp
					, cPrmId
					, RtT_TwrNumer lTwrNumer
					, lPrmDataOd
					, lPrmDataDo
					, cWartosc
					, cWartWaluta
					, lRodzajCeny
					, bCykliczna
					, cNettoBrutto
					, dProgWartosc
					, lRodzajDok
					, lRodzajUpustu
					, bPakietowa
					, lPrmTyp
					, bJestDPR
					, bProgTyp
					, lCyklRodzaj
					, lCyklCzestotliwosc
					, lCyklWystepowanie
					, lCyklDniOd
					, lCyklDniDo
				FROM UpustyRetro
				INNER JOIN cdn.RetroTowary ON RtT_RtKId = cPrmId
				INNER JOIN cdn.TwgView ON RtT_TwrNumer = TGV_GIDNumer
					AND RtT_TwrTyp = TGV_GIDTyp
				WHERE lPPoziom < 2
				)
		SELECT *
		FROM UpustyRetro
		)

        

          /* [PobierzUpustyRetroDlaTwr] */

CREATE FUNCTION [CDN].[PobierzUpustyRetroDlaTwr] (@p_TwrNumer INT)
RETURNS TABLE
AS
RETURN (
		WITH UpustyRetro AS (
				SELECT cast(RtK_Nazwa AS VARCHAR(201)) cPrmNazwa
					, cast(RtK_Nazwa AS VARCHAR(201)) cNazwa
					, 1 lPPoziom
					, - 1 lPrmLp
					, cast(0 AS SMALLINT) lTyp
					, RtK_Id cPrmId
					, 0 lKntNumer
					, CASE 
						WHEN RtK_DataOd < 93890
							THEN 86400 * (RtK_DataOd - 69035)
						ELSE 2147483647
						END lDataOd
					, CASE 
						WHEN RtK_DataDo < 93890
							THEN 86400 * (RtK_DataDo - 69035)
						ELSE 2147483647
						END lDataDo
					, CONVERT(VARCHAR(20), RtP_Rabat) AS cWartosc
					, '' cWartWaluta
					, RtK_RodzajCeny lRodzajCeny
					, 0 bCykliczna
					, CASE 
						WHEN RtP_TypRabatu = 1
							THEN 'Netto'
						ELSE ''
						END cNettoBrutto
					, RtP_WartoscProgu dProgWartosc
					, 1 lRodzajDok
					, RtP_TypRabatu lRodzajUpustu
					, 0 bPakietowa
					, 0 lCyklRodzaj
					, 0 lCyklCzestotliwosc
					, 0 lCyklWystepowanie
					, 0 lCyklDniOd
					, 0 lCyklDniDo
				FROM cdn.RetroTowary
				INNER JOIN cdn.RetroKarty ON RtK_Id = RtT_RtKId
				OUTER APPLY (
					SELECT TOP 1 *
					FROM cdn.RetroProgi
					WHERE RtP_RtKId = RtK_Id
					ORDER BY RtP_WartoscProgu ASC
					) NajmniejszyRetroProg -- ON NajmniejszyRetroProg.RtP_RtKId = RtK_Id
				WHERE RtT_TwrNumer = @p_TwrNumer
					AND RtT_TwrTyp = 16
					OR RtT_TwrNumer IN (
						SELECT TwL_GrONumer
						FROM cdn.TwrLinki
						WHERE TwL_GIDNumer = @p_TwrNumer
							AND TwL_GIDTyp = 16
						)
				
				UNION ALL
				
				SELECT CASE 
						WHEN RKt_KntNumer = 0
							AND RKt_KntTyp = - 32
							THEN cast('Grupa główna' AS VARCHAR(201))
						ELSE cast(KGD_Kod AS VARCHAR(201))
						END cPrmNazwa
					, cNazwa cNazwa
					, lPPoziom + 1 lPPoziom
					, - 1 lPrmLp
					, RKt_KntTyp lTyp
					, cPrmId
					, RKt_KntNumer lKntNumer
					, lDataOd
					, lDataDo
					, cWartosc
					, cWartWaluta
					, lRodzajCeny
					, bCykliczna
					, cNettoBrutto
					, dProgWartosc
					, lRodzajDok
					, lRodzajUpustu
					, bPakietowa
					, lCyklRodzaj
					, lCyklCzestotliwosc
					, lCyklWystepowanie
					, lCyklDniOd
					, lCyklDniDo
				FROM UpustyRetro
				INNER JOIN cdn.RetroKontrahenci ON RKt_RtKId = cPrmId
				INNER JOIN cdn.KntGrupyDom ON (
						RKt_KntNumer = KGD_GIDNumer
						AND RKt_KntTyp = KGD_GIDTyp
						)
					OR (
						RKt_KntNumer = 0
						AND RKt_KntTyp = - 32
						)
				WHERE lPPoziom < 2
				)
		SELECT *
		FROM UpustyRetro
		)
