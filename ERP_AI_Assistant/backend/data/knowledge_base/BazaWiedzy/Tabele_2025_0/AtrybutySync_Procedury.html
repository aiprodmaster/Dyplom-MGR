/* XLSynchroUsunKlaseAtrybutu */

CREATE PROCEDURE CDN.XLSynchroUsunKlaseAtrybutu (@_OptimaID INT) 
AS
BEGIN
	-- TFSID:195830
		IF @_OptimaID = 0 RETURN
        UPDATE CDN.AtrybutyKlasy SET Atk_Synchronizowany = 0, AtK_OptimaID = 0 WHERE Atk_OptimaId = @_OptimaID

END

        

          /* XLSynchroUsunGrupeKlasAtrybutow */

CREATE PROCEDURE CDN.XLSynchroUsunGrupeKlasAtrybutow (@_OptimaID INT) 
AS
BEGIN
	IF @_OptimaID = 0 RETURN
	UPDATE CDN.AtrybutyKomplety SET AKp_OptimaId = 0 WHERE AKp_OptimaId = @_OptimaID
END

        

          /* XLSynchroUsunAtrybut */

CREATE PROCEDURE CDN.XLSynchroUsunAtrybut (@_OptimaID INT) 
AS
BEGIN
	IF @_OptimaID = 0 RETURN
	DELETE FROM CDN.Atrybuty WHERE Atr_OptimaId = @_OptimaID
END

        

          /* XLSynchroUsunAtrybutyHist */

CREATE PROCEDURE CDN.XLSynchroUsunAtrybutyHist (@_OptimaID INT) 
AS
BEGIN
	IF @_OptimaID = 0 RETURN
	DELETE FROM CDN.AtrybutyHist WHERE AtH_ID IN (SELECT Atr_ID FROM CDN.Atrybuty WHERE Atr_OptimaId = @_OptimaID)
END
