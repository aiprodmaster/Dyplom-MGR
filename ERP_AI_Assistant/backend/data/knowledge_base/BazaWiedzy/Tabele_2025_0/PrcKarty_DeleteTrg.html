/* PrcKarty_SprawdzSynchronizacje */

CREATE TRIGGER CDN.PrcKarty_SprawdzSynchronizacje
  ON CDN.PrcKarty
  FOR DELETE
AS
BEGIN
  SET NOCOUNT ON
  IF EXISTS(SELECT * FROM DELETED WHERE DELETED.Prc_ZewnetrznySys = 1)
  BEGIN
    RAISERROR('#CDN_INFO/##CDN_1=Informacje o pracowniku są współdzielone z oddziałami. Nie można usunąć pracownika./#',16,1)
	ROLLBACK
  END
  SET NOCOUNT OFF
END
