/* DataClarionPrzesun */

CREATE FUNCTION CDN.DataClarionPrzesun(@DataPrzed int,@IloscJednostek int,@Jednostka int)

returns int
as
begin
declare @DataPo int

	if @Jednostka = 0
		set @DataPo  = datediff(dd,CONVERT(DATETIME,'18001228',11),DATEADD(dd,@IloscJednostek,DATEADD( dd, @DataPrzed,CONVERT(DATETIME,'18001228',11)))) 
	else
		if @Jednostka = 1
			set @DataPo  =  datediff(dd,CONVERT(DATETIME,'18001228',11),DATEADD(wk,@IloscJednostek,DATEADD( dd, @DataPrzed,CONVERT(DATETIME,'18001228',11))))
		else
			if @Jednostka = 2
				set @DataPo  =  datediff(dd,CONVERT(DATETIME,'18001228',11),DATEADD(mm,@IloscJednostek,DATEADD( dd, @DataPrzed,CONVERT(DATETIME,'18001228',11))))
			else
				if @Jednostka = 3
					set @DataPo  =  datediff(dd,CONVERT(DATETIME,'18001228',11),DATEADD(yy,@IloscJednostek,DATEADD( dd, @DataPrzed,CONVERT(DATETIME,'18001228',11))))						
				else
					set @DataPo  = @DataPrzed
return @DataPo 
end
