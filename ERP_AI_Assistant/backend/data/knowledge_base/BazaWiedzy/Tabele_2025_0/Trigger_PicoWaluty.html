/* [PicoWaluty_CzasModyfikacji] */

CREATE TRIGGER [CDN].[PicoWaluty_CzasModyfikacji]
ON [CDN].[PicoWaluty]
FOR INSERT, UPDATE
AS
BEGIN
SET NOCOUNT ON
        UPDATE O SET O.PwL_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
        FROM CDN.PicoWaluty O JOIN Inserted I ON I.pwl_Id = O.PWl_ID
SET NOCOUNT OFF
END
