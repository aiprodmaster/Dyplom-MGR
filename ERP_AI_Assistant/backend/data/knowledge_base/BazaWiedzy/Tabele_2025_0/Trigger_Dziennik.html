/* Dziennik_AktualizujDateZapisu */

create trigger cdn.Dziennik_AktualizujDateZapisu
on CDN.Dziennik
for UPDATE
as
begin

	if Update(DZK_Dzien) or Update(DZK_Miesiac) or Update(DZK_Rok)
	begin
		update cdn.Dekrety set
			Dt_Dzien = Dzk_Dzien,
			Dt_Miesiac = Dzk_Miesiac,
			Dt_Rok = Dzk_Rok
		from inserted join cdn.Dekrety on Dt_GIDNumer = Dzk_GIDNumer

	end
end;
