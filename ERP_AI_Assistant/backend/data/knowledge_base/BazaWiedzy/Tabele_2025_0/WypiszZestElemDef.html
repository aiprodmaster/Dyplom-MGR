/* WypiszZestElemDef */

CREATE FUNCTION CDN.WypiszZestElemDef(@GIDNumer INT,@GIDTyp SMALLINT)
RETURNS @t1 TABLE(ZKE_GIDNumer INT,ZKE_GIDLp SMALLINT ,ZKE_Pozycja1 SMALLINT,ZKE_Pozycja2 SMALLINT,
ZKE_Pozycja3 SMALLINT,ZKE_Pozycja4 SMALLINT,ZKE_Pozycja5 SMALLINT,ZKE_Pozycja6 SMALLINT,ZKE_Pozycja7 SMALLINT,
ZKE_Pozycja8 SMALLINT,ZKE_Pozycja9 SMALLINT,ZKE_Pozycja10 SMALLINT,ZKD_Wartosc VARCHAR(40))
AS
BEGIN
	-- Add the SELECT statement with parameter references here
	INSERT INTO @t1 select ZKE_GIDNumer,ZKE_GIDLp,ZKE_Pozycja1,ZKE_Pozycja2,ZKE_Pozycja3,ZKE_Pozycja4,ZKE_Pozycja5,
	ZKE_Pozycja6,ZKE_Pozycja7,ZKE_Pozycja8,ZKE_Pozycja9,ZKE_Pozycja10,ISNULL(ZKD_Wartosc,'') as ZKD_Wartosc from cdn.ZestKSElem
	left outer join cdn.ZestKSDef on ZKE_GIDNumer=ZKD_GIDNumer AND ZKE_GIDLp=ZKD_GIDLp
	where ZKE_GIDTyp=2784 and ZKE_GIDNumer=@GIDNumer
	order by ZKE_Pozycja1,ZKE_Pozycja2,ZKE_Pozycja3,ZKE_Pozycja4,ZKE_Pozycja5,
	ZKE_Pozycja6,ZKE_Pozycja7,ZKE_Pozycja8,ZKE_Pozycja9,ZKE_Pozycja10
RETURN 
END
