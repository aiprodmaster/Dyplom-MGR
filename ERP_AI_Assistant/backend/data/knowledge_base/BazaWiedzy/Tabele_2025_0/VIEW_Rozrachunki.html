/* Rozrachunki */

CREATE VIEW CDN.Rozrachunki ( ROZ_GIDTyp, ROZ_GIDFirma, ROZ_GIDNumer, ROZ_GIDLp, 
  ROZ_TrpTyp, ROZ_TrpFirma, ROZ_TrpNumer, ROZ_TrpLp, 
  ROZ_KAZTyp, ROZ_KAZFirma, ROZ_KAZNumer, ROZ_KAZLp, 
  ROZ_DtTyp, ROZ_DtFirma, ROZ_DtNumer, ROZ_DtLp, 
  ROZ_CtTyp, ROZ_CtFirma, ROZ_CtNumer, ROZ_CtLp, 
  ROZ_Kwota, ROZ_KwotaSys, ROZ_DataOddzialu, ROZ_DataCentrali, 
  ROZ_Waluta, ROZ_DTDC, ROZ_CTDC, 
  ROZ_OpeTypRL, ROZ_OpeFirmaRL, ROZ_OpeNumerRL, ROZ_OpeLpRL, 
  ROZ_OpeTypRZ, ROZ_OpeFirmaRZ, ROZ_OpeNumerRZ, ROZ_OpeLpRZ, 
  ROZ_RK, ROZ_RKStrona, ROZ_DataRozliczenia, ROZ_DataRozrachunku, ROZ_ParID, ROZ_Wycena) 
  AS SELECT 433, R2_GIDFirma, R2_ID, 1,
  CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Typ END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Numer END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Lp END,
  CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Typ END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Numer END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Lp END,
  CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE 432 END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1Numer END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1Lp END,
  CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE 432 END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2Numer END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2Lp END,
  R2_KwotaWal1, R2_KwotaSys, 0, 0,
  R2_Waluta1, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1DC END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2DC END,
  CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE 128 END, CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE R2_OpeNumerRL END, 0,
  CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE 128 END, CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE R2_OpeNumerRZ END, 0,
  CASE WHEN ISNULL(R2_RK,0)=0 THEN 0 ELSE R2_RK END, CASE WHEN ISNULL(R2_RKStrona,0)=0 THEN 0 ELSE R2_RKStrona END, CASE WHEN ISNULL(R2_DataRozliczenia,0)=0 THEN 0 ELSE R2_DataRozliczenia END, CASE WHEN ISNULL(R2_DataRozrachunku,0)=0 THEN 0 ELSE R2_DataRozrachunku END,
  R2_ParID, R2_Wycena
  FROM CDN.Rozliczenia
  WHERE R2_Wycena = 0
  UNION ALL SELECT 433, R2_GIDFirma, R2_ID, 2,
  CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Typ END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Numer END, CASE WHEN ISNULL(R2_Dok2Numer,0)=0 THEN 0 ELSE R2_Dok2Lp END,
  CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Typ END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Numer END, CASE WHEN ISNULL(R2_Dok1Numer,0)=0 THEN 0 ELSE R2_Dok1Lp END,
  CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE 432 END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2Numer END, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2Lp END,
  CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE 432 END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1Numer END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1Lp END,
  R2_KwotaWal2, R2_KwotaSys, 0, 0,
  R2_Waluta2, CASE WHEN ISNULL(R2_Dekret2Numer,0)=0 THEN 0 ELSE R2_Dekret2DC END, CASE WHEN ISNULL(R2_Dekret1Numer,0)=0 THEN 0 ELSE R2_Dekret1DC END,
  CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE 128 END, CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_OpeNumerRL,0) = 0 THEN 0 ELSE R2_OpeNumerRL END, 0,
  CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE 128 END, CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE R2_GIDFirma END, CASE WHEN ISNULL(R2_OpeNumerRZ,0) = 0 THEN 0 ELSE R2_OpeNumerRZ END, 0,
  CASE WHEN ISNULL(R2_RK,0)=0 THEN 0 ELSE R2_RK END, CASE WHEN ISNULL(R2_RKStrona,0)=0 THEN 0 ELSE R2_RKStrona END, CASE WHEN ISNULL(R2_DataRozliczenia,0)=0 THEN 0 ELSE R2_DataRozliczenia END, CASE WHEN ISNULL(R2_DataRozrachunku,0)=0 THEN 0 ELSE R2_DataRozrachunku END,
  R2_ParID, R2_Wycena
  FROM CDN.Rozliczenia
  WHERE R2_Wycena = 0
