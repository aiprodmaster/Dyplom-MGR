/* [UmwElem_DeletePrzenumeruj] */

CREATE TRIGGER [CDN].[UmwElem_DeletePrzenumeruj]
   ON  [CDN].[UmwElem]
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.UmwElem SET UmE_Pozycja = E.UmE_Pozycja - 1
FROM CDN.UmwElem E
INNER JOIN Deleted D ON E.UmE_GIDTyp = D.UmE_GIDTyp AND E.UmE_GIDNumer = D.UmE_GIDNumer
WHERE D.UmE_Pozycja < E.UmE_Pozycja

SET NOCOUNT OFF
END
