/* MobBI_EksportKonfig */

CREATE PROCEDURE CDN.MobBI_EksportKonfig
  @OpeIdent nvarchar(8)
AS 
    DECLARE @UserAccount varchar(100)
    SELECT @UserAccount = CASE WHEN ISNULL(Ope_NTKonto,'')='' THEN Prc_NTKonto ELSE Ope_NTKonto END FROM CDN.OpeKarty LEFT JOIN CDN.PrcKarty ON Ope_PrcNumer = Prc_GIDNumer WHERE Ope_Ident = @OpeIdent
	DECLARE @MetaDb varchar(100)
    SELECT @MetaDb = Kon_Wartosc FROM CDN.Konfig WHERE Kon_Numer = 5005
	DECLARE @MetaSrv varchar(100)
    SELECT @MetaSrv = Kon_Wartosc FROM CDN.Konfig WHERE Kon_Numer = 5011

    SELECT @MetaSrv As MetaServer, @MetaDb As MetaDatabase, @UserAccount As UserAccount
