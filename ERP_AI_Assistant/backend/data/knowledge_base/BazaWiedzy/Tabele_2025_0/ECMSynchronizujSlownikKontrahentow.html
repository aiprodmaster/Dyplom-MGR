/* ECMSynchronizujSlownikKontrahentow */

CREATE FUNCTION CDN.ECMSynchronizujSlownikKontrahentow()
RETURNS TABLE
AS
RETURN
(
	SELECT Knt_Akronim as Akronim, Knt_Nazwa1 as Nazwa,
	Knt_Ean as Ean,  Knt_Ulica as Ulica,  Knt_Kraj as Kraj,
	Knt_Miasto as Miasto,  Knt_KodP as Kod_Pocztowy,
	Knt_Wojewodztwo as Wojewodztwo,
	Knt_Powiat as Powiat,  Knt_Gmina as Gmina,
	Knt_NIP as NIP, Knt_Regon as Regon, Knt_Pesel as Pesel
	FROM CDN.KntKarty
);
