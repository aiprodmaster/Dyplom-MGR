/* [InwMag_DeletePrzenumeruj] */

CREATE TRIGGER [CDN].[InwMag_DeletePrzenumeruj]
   ON  [CDN].[InwMag]
   FOR DELETE
AS
BEGIN
SET NOCOUNT ON;

UPDATE CDN.InwMag SET InM_Pozycja = I.InM_Pozycja - 1
FROM CDN.InwMag I
INNER JOIN Deleted D ON I.InM_InNId = D.InM_InNId
WHERE D.InM_Pozycja < I.InM_Pozycja

SET NOCOUNT OFF
END
