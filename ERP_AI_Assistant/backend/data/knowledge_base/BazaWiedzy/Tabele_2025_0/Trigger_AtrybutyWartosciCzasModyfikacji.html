/* [AtrybutyWartosci_CzasModyfikacji] */

CREATE TRIGGER [CDN].[AtrybutyWartosci_CzasModyfikacji]
ON [CDN].[AtrybutyWartosci]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.atw_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.AtrybutyWartosci N join inserted I on N.Atw_ID = I.Atw_ID

	SET NOCOUNT OFF
END
