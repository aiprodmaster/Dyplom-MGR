/* [SekMailInfo] */

CREATE VIEW [CDN].[SekMailInfo]
AS
SELECT     SDS_Id,
                          (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig
                            WHERE      (Kon_Numer = 2111)) AS SerwerSmtp,
                                     (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig
                            WHERE      (Kon_Numer = 2123)) AS Port,
                                     (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig
                            WHERE      (Kon_Numer = 2122)) AS AdresPotwierdzajacy,
                                     (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig
                            WHERE      (Kon_Numer = 2115)) AS MailOdNazwa,
                          (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig AS Konfig_5
                            WHERE      (Kon_Numer = 2112)) AS Uzytkownik,
                          (SELECT     Kon_Wartosc
                           FROM          CDN.Konfig AS Konfig_4
                            WHERE      (Kon_Numer = 2113)) AS Haslo,
                          (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig AS Konfig_3
                            WHERE      (Kon_Numer = 2114)) AS MailOd, 
                                         SDS_EmailTemat, SDS_EmailTresc,SDS_Nazwa,
                          (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig AS Konfig_2
                            WHERE      (Kon_Numer = 7206)) AS Url,
                          (SELECT     Kon_Wartosc
                            FROM          CDN.Konfig AS Konfig_1
                            WHERE      (Kon_Numer = 7202)) AS CzyWysylac
FROM         CDN.SekDokStatusy;
