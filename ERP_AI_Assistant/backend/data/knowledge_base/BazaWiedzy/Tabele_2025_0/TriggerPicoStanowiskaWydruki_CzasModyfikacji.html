/* [PicoStanowiskaWydruki_CzasModyfikacji] */

CREATE TRIGGER [CDN].[PicoStanowiskaWydruki_CzasModyfikacji]
ON [CDN].[PicoStanowiskaWydruki]
FOR INSERT, UPDATE
AS
BEGIN
SET NOCOUNT ON
	UPDATE O SET O.Pwr_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.PicoStanowiskaWydruki O JOIN Inserted I ON I.pwr_Id = O.PWr_ID
SET NOCOUNT OFF
END
