/* PrzeliczPunktyLojalNazwaBudzetu */

CREATE PROCEDURE CDN.PrzeliczPunktyLojalNazwaBudzetu
AS
 select TOP 1 'PUNKTY LOJALNOŚCIOWE '+CAST(CAST(SUBSTRING(PRM_Akronim,22,10) AS INT)+1 AS VARCHAR)
  from (select PRM_AKRONIM, PRM_Id from cdn.PrmKarty union select 'PUNKTY LOJALNOŚCIOWE 0',0) a where PRM_Akronim like 'PUNKTY LOJALNOŚCIOWE %' ORDER BY PRM_ID DESC
