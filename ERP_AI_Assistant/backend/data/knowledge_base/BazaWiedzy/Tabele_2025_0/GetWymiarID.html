/* GetWymiarID */

CREATE FUNCTION CDN.GetWymiarID (@WMRID INT)  
RETURNS INT AS  
BEGIN 
Declare @ElementID INT; 
DECLARE @ParentID INT;

SELECT @ElementID=WMR_ID, @ParentID = WMR_ParID
FROM CDN.wymiary WHERE WMR_ID=@WmrID

WHILE (@ParentID>0)
	BEGIN 
		SET @ElementID=@ParentID 
		SELECT @ElementID=WMR_ID, @ParentID = WMR_ParID
		FROM CDN.wymiary WHERE WMR_ID=@ElementID
	END 
RETURN(@ElementID)
END
