/* POS_AktualizacjaStatusuPlatnosci */

CREATE PROCEDURE CDN.POS_AktualizacjaStatusuPlatnosci
	@PaymentStatusId INT,
	@RealizationStatusId INT,
	@GIDNumer INT
AS
BEGIN
SET NOCOUNT ON;
UPDATE CDN.ZamNag SET ZaN_StatusPlatnosciPOS = @PaymentStatusId,ZAN_StatusRealizacji=@RealizationStatusId WHERE ZaN_GIDNumer = @GIDNumer
SET NOCOUNT OFF;	
END
