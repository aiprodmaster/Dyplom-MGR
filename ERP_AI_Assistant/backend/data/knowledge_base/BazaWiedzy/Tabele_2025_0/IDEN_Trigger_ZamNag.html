/* ZamNag_InitFields */

Create Trigger CDN.ZamNag_InitFields on CDN.ZamNag for insert
as 
Begin 

	update T 
	set t.ZaN_ZrdTyp   = t.ZAN_GIDTyp,
	t.ZaN_ZrdFirma 	 = t.ZaN_GIDFirma ,
	t.ZaN_ZrdNumer 	 = t.ZaN_GIDNumer ,
	t.ZaN_ZrdLp 	 = t.ZaN_GIDLp
	From CDN.ZamNag T join inserted on 
		 t.ZaN_GIDNumer = inserted.ZaN_GIDNumer
	Where isnull(inserted.ZaN_ZrdTyp,0) = 0
End
