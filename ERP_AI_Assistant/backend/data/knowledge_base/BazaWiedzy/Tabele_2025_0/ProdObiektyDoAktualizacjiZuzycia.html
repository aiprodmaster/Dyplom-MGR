/* [ProdObiektyDoAktualizacjiZuzycia] */

CREATE FUNCTION [CDN].[ProdObiektyDoAktualizacjiZuzycia](@POBId VARCHAR(8000))
RETURNS TABLE
AS
RETURN
SELECT POB_Id FROM CDN.ProdObiekty JOIN CDN.ProdZuzycieNarzedzia ON PZN_POBId = POB_Id
WHERE POB_RejestrowacZuzycie = 1 and PZN_SposobObliczaniaZ > 0 AND POB_Id IN(SELECT Num FROM CDN.Str2List(@POBId))
