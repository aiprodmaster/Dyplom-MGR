/* [Nazwy_CzasModyfikacji] */

CREATE TRIGGER [CDN].[Nazwy_CzasModyfikacji]
ON [CDN].[Nazwy]
FOR INSERT, UPDATE
AS
BEGIN
	SET NOCOUNT ON

	UPDATE N SET N.Naz_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.Nazwy N join inserted I on N.Naz_GIDTyp = I.Naz_GIDTyp and N.Naz_GIDLp = I.Naz_GIDLp

	SET NOCOUNT OFF
END
