/* KazTrpView */

CREATE VIEW CDN.KazTrpView (GIDTyp, GIDFirma, GIDNumer, GIDLp, Kwota, Termin, Data) 
AS 
SELECT kaz_gidtyp, kaz_gidfirma, kaz_gidnumer, kaz_gidlp, kaz_kwota, kaz_datazapisu, kaz_datazapisu 
FROM CDN.zapisy 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, trn_data2 
FROM CDN.traplat JOIN CDN.tranag ON trn_gidtyp=trp_gidtyp and trn_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, imn_datawystawienia 
FROM CDN.traplat JOIN CDN.impnag ON imn_gidtyp=trp_gidtyp and imn_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, men_datawprow 
FROM CDN.traplat JOIN CDN.memnag ON men_gidtyp=trp_gidtyp and men_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, pin_data
FROM CDN.traplat JOIN CDN.piknag ON pin_gidtyp=trp_gidtyp and pin_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, san_datazgloszenia 
FROM CDN.traplat JOIN CDN.sadnag ON san_gidtyp=trp_gidtyp and san_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, upn_dataup 
FROM CDN.traplat JOIN CDN.uponag ON upn_gidtyp=trp_gidtyp and upn_gidnumer=trp_gidnumer 
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, bos_data
FROM CDN.traplat JOIN CDN.bilansotwarciaselem ON bos_trptyp=trp_gidtyp and bos_trpnumer=trp_gidnumer and bos_trplp=trp_gidlp
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, kmn_datadokumentu
FROM CDN.traplat JOIN CDN.KompNag ON trp_gidtyp=434 and kmn_id=trp_gidnumer
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, krp_datazamkniecia
FROM CDN.traplat JOIN CDN.Raporty ON krp_gidtyp=trp_gidtyp and krp_gidnumer=trp_gidnumer
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, lpl_datawyp
FROM CDN.traplat join cdn.ListyPlac on trp_gidtyp=lpl_typ join cdn.Wyplaty on trp_gidnumer=Wpl_id and WPL_LPLId = LPL_Id
UNION ALL 
SELECT trp_gidtyp, trp_gidfirma, trp_gidnumer, trp_gidlp, trp_kwota, trp_termin, datediff(dd,'18001228',cdn.TSToDate(KBNN_DataWystawienia, 1))
FROM CDN.traplat JOIN CDN.KBNNag ON trp_gidtyp=4608 and KBNN_Id=trp_gidnumer
