/* [TwrDost_DeleteTKKHandler] */

CREATE TRIGGER [CDN].[TwrDost_DeleteTKKHandler]
  ON [CDN].[TwrDost]
  FOR DELETE
AS
SET NOCOUNT ON
   delete a from deleted d join cdn.TwrKodyKnt a on d.TWD_KlasaKnt=a.TKK_KlasaKnt and d.TWD_KntNumer=a.TKK_KntNumer join CDN.TwrKody b on a.TKK_TwKId=b.TwK_Id and d.TWD_TwrNumer = b.TwK_TwrNumer
SET NOCOUNT OFF
