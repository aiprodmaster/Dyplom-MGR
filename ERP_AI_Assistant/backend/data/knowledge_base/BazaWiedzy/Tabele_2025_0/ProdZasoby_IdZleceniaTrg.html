/* ProdZasoby_IdZlecenia */

CREATE TRIGGER CDN.ProdZasoby_IdZlecenia
  ON CDN.ProdZasoby
  AFTER INSERT
AS
	SET NOCOUNT ON
	UPDATE CDN.ProdZasoby SET PZA_PZLId = PZLId
	FROM (SELECT PZA_Id Id,PPC_Zlecenie PZLId FROM inserted i JOIN CDN.ProdCzynnosci ON PCZ_Id = I.PZA_Czynnosc JOIN CDN.ProdProcesy ON PPC_Id = PCZ_Proces) Z WHERE Z.Id = PZA_Id
	
	SET NOCOUNT OFF
