/* PicoStanowiska_DataDodania */

CREATE  TRIGGER CDN.PicoStanowiska_DataDodania
  ON CDN.PicoStanowiska
  FOR Insert, Update
AS
 BEGIN
	   UPDATE N SET N.PSa_DataDodania = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
		FROM CDN.PicoStanowiska N join inserted I on N.PSa_ID = I.PSa_ID AND I.PSa_DataDodania = 0
 END
