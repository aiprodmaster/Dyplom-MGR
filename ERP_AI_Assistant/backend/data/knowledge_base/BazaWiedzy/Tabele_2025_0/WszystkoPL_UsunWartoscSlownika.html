/* [WszystkoPL_UsunWartoscSlownika] */

CREATE  PROCEDURE [CDN].[WszystkoPL_UsunWartoscSlownika]
@WszystkoPLSloId  INT, -- ID slownika z wszystko.pl
@CategoryId INT
AS
SET NOCOUNT ON

DECLARE @KspkId INT
SET @KspkId = (select KSPK_Id from cdn.KgtSlownikiParametruKategorii
				join cdn.kgtslownik on KSPK_SlownikId = KSlo_Id
				join cdn.KgtParametryKategorii on KPK_Id = KSPK_ParametrKategoriiId
				WHERE KSlo_WszystkoPLId = @WszystkoPLSloId AND KPK_KategoriaId = @CategoryId);

DELETE FROM cdn.KgtSlownikiParametruKategorii WHERE KSPK_Id = @KspkId
