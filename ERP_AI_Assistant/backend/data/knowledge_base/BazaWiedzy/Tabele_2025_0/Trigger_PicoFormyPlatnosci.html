/* [PicoStanowiskaFormyPl_CzasModyfikacji] */

CREATE TRIGGER [CDN].[PicoStanowiskaFormyPl_CzasModyfikacji]
ON [CDN].[PicoStanowiskaFormyPl]
FOR INSERT, UPDATE
AS
BEGIN
SET NOCOUNT ON
        UPDATE O SET O.PFP_CzasModyfikacji = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
        FROM CDN.PicoStanowiskaFormyPl O JOIN Inserted I ON I.PFP_PSaID = O.PFP_PSaID AND I.PFP_FormaNr = O.PFP_FormaNr
SET NOCOUNT OFF
END
