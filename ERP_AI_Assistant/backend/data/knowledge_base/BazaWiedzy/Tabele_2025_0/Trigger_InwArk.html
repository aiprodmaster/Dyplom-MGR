/* InwArk_CzasModyfikacji */

CREATE TRIGGER CDN.InwArk_CzasModyfikacji
ON CDN.InwArk
FOR INSERT, UPDATE
AS
BEGIN
	 IF Context_Info() = 0x15ff2808
	 RETURN

  UPDATE CDN.InwArk SET InA_CzasModyfikacji = datediff(s,convert(datetime,'1990-01-01',120),getdate())
  FROM CDN.InwArk N JOIN inserted I ON N.InA_Id = I.InA_Id
END
