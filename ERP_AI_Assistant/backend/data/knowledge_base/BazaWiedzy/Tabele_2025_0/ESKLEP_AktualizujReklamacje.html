/* ESKLEP_AktualizujReklamacje */

CREATE PROCEDURE CDN.ESKLEP_AktualizujReklamacje
	@ZaN_GIDTyp Int,
	@ZaN_ZamTyp Int,
	@ZaN_Rodzaj Int,
	@ZaN_ZamNumer Int,
	@ZaN_ZamRok Int,
	@ZaN_ZamSeria VarChar(256),
	@ZaN_ZamMiesiac Int,
	@Id Int

AS
SET NOCOUNT ON

UPDATE CDN.ReklElem SET RLE_ZrdDokumentObcy = CDN.NumerDokumentu(CDN.DokMapTypDokumentu(@ZaN_GIDTyp, @ZaN_ZamTyp, @ZaN_Rodzaj), 0, 0, @ZaN_ZamNumer, @ZaN_ZamRok, @ZaN_ZamSeria, @ZaN_ZamMiesiac)
WHERE RLE_RLNId = @Id

SET NOCOUNT OFF
