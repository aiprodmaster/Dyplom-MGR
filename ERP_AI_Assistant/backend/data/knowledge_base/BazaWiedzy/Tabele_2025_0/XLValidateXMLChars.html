/* XLValidateXMLChars */

CREATE FUNCTION CDN.XLValidateXMLChars (@_sIN VARCHAR(MAX))
RETURNS VARCHAR(MAX)
AS
BEGIN
	DECLARE @sResult VARCHAR(MAX)
	DECLARE @i INT
	DECLARE @sChar CHAR
	
	DECLARE @sInvalidChars VARCHAR(3)
	SET @sInvalidChars = CHAR(0x3B)+CHAR(0x3C)+CHAR(0x3E)

	SET @i = 1
	SET @sResult = ''
	
	WHILE @i<=len(@_sIN)
	BEGIN
		SET @sChar = SUBSTRING(@_sIN, @i, 1)
		IF ASCII(@sChar) >= 0x20
			SET @sResult = @sResult+@sChar		
		SET @i = @i+1
	END

	RETURN @sResult
END
