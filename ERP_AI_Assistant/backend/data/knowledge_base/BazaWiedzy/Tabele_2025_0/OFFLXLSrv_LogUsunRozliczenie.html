/* OFFLXLSrv_LogUsunRozliczenie */

CREATE PROCEDURE CDN.OFFLXLSrv_LogUsunRozliczenie
@ID INT,
@ZewnetrznyID INT,
@ZewnetrznySys INT,
@ZewnetrznyTyp INT
AS
	
	IF @ZewnetrznyTyp NOT IN (1,2,3)
		RETURN 0
		
	DECLARE @rv			INT
			,@PcKId		INT
			,@ObiTyp	SMALLINT
			,@ObiNumer	INT
			,@ObiektTyp	INT
			,@ObiektId	INT
	SELECT	@PcKId = @ZewnetrznySys
			,@ObiTyp = CASE @ZewnetrznyTyp WHEN 1 THEN -433 ELSE -434 END
			,@ObiNumer = @ID
			,@ObiektTyp = CASE @ZewnetrznyTyp WHEN 2 THEN 7684 WHEN 3 THEN 784 ELSE 0 END
			,@ObiektId = @ZewnetrznyID
			
	EXEC @rv = CDN.OFFLXLSrv_LogUsunObiekt @PckId, @ObiTyp, @ObiNumer, @ObiektTyp, @ObiektId
	
	RETURN @rv
