/* [WebPOS_ZapiszDateSynchronizacji] */

CREATE PROCEDURE [CDN].[WebPOS_ZapiszDateSynchronizacji]

AS
BEGIN
SET NOCOUNT ON;
	
	UPDATE CDN.PicoStanowiska set PSa_LastSync=DateDiff(d, convert(datetime,'28-12-1800',105),getdate()) where PSa_TypAplikacji = 2
	UPDATE a set pck_stan = 2,PcK_TSSynchr=DATEDIFF(second, convert(datetime,'01-01-1990',105),convert(datetime, getdate(), 105)) 
		from CDN.PicoKonfig a join cdn.picostanowiska on psa_pckid=pck_picoid  where PSa_TypAplikacji = 2
	
	SET NOCOUNT OFF;	
END
