/* [CRMWzorceWizyt_CzasModyfikacjiC] */

CREATE TRIGGER [CDN].[CRMWzorceWizyt_CzasModyfikacjiC]
   ON  [CDN].[CRMWzorceWizyt]
   FOR INSERT, UPDATE
AS 
BEGIN
SET NOCOUNT ON;

	UPDATE E
	SET E.CWW_CzasModyfikacjiC = DATEDIFF(s,CONVERT(DATETIME,'1990-01-01',120),GETDATE())
	FROM CDN.CRMWzorceWizyt E, Inserted
	WHERE E.CWW_Id = Inserted.CWW_Id      

SET NOCOUNT OFF
END
