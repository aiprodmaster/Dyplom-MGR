/* [PodziePole] */

CREATE FUNCTION [CDN].[PodziePole] (@Pole nvarchar(256),@separator nvarchar(1))
RETURNS @podzielone TABLE(opis nvarchar(max))
as
begin	  
	 while CHARINDEX(@separator,@Pole) > 0
	 begin	
		insert into @podzielone select left(@Pole,CHARINDEX(@separator,@Pole)-1)
		select @Pole = right(@Pole,len(@Pole)-CHARINDEX(@separator,@Pole))
	 end
	 IF len(@Pole) > 0
		insert into @podzielone select @Pole
	 return
end
