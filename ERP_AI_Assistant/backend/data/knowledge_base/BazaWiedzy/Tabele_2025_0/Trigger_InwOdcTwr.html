/* [InwOdcTwr_DeletePrzenumeruj] */

CREATE TRIGGER [CDN].[InwOdcTwr_DeletePrzenumeruj]
   ON  [CDN].[InwOdcTwr]
   FOR DELETE
AS 
BEGIN
SET NOCOUNT ON;

UPDATE CDN.InwOdcTwr SET IOT_InOLp = I.IOT_InOLp - 1
FROM CDN.InwOdcTwr I
INNER JOIN Deleted D ON I.IOT_InOId = D.IOT_InOId
WHERE D.IOT_InOLp < I.IOT_InOLp

SET NOCOUNT OFF
END
