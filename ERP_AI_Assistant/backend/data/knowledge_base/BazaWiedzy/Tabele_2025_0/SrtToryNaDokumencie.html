/* SrtToryNaDokumencie */

CREATE function CDN.SrtToryNaDokumencie(@She_GIDNumer int) 
returns smallint
AS 
BEGIN 
	
	return(
		select case when max(SHE_TorUm_Lp) > 0 then 1 else 0 end +
			case when max(SHE_TorAm_Lp) > 0 then 2 else 0 end +
			case when max(SHE_Tor3_Lp) > 0 then 4 else 0 end +
			case when max(SHE_Tor4_Lp) > 0 then 8 else 0 end +
			case when max(SHE_Tor5_Lp) > 0 then 16 else 0 end ToryBit
		from cdn.SrtHistElem where SHE_GIDNumer= @She_GIDNumer
		group by SHE_GIDNumer	
	)

END
