/* ProdCzynnosci_AktualizujPKJN */

CREATE TRIGGER CDN.ProdCzynnosci_AktualizujPKJN
  ON CDN.ProdCzynnosci
  FOR DELETE
AS
begin
SET NOCOUNT ON
	UPDATE CDN.ProdPKJNag SET PKJN_PCZId = 0 FROM deleted d WHERE PKJN_PCZId = d.PCZ_Id
SET NOCOUNT OFF
end
